파라메트릭 서치
```
#include <stdio.h>
#include <stdlib.h>

int a[1000000];
int N, M;

int solve() {
    long c;

    int st, en, i, m;
    st = 0;
    en = 1000000000;

    while (st < en) {
        m = ((long)st + en + 1)/2;

        for (c=0, i=0; i<N; i++) {
            if (a[i] > m) {
                c += a[i] - m;
            }
        }

        if (c >= M) {
            st = m;
        } else {
            en = m-1;
        }
    }

    return st;
}

int main() {
    int i;
    scanf("%d %d", &N, &M);

    for (i=0; i<N; i++) {
        scanf("%d", &a[i]);
    }

    printf("%d\n", solve());

    return 0;
}
```
