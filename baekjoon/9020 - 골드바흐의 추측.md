소수를 미리 테이블에 채워넣고 사용하였다.
[[1644 - 소수의 연속합]]에서 쓴 소수 판별법 사용.


```c
#include <stdio.h>

int isPrime(int n)
{
    if (n == 2 || n == 3)
        return 1;

    if (n <= 1 || n % 2 == 0 || n % 3 == 0)
        return 0;

    for (int i = 5; i * i <= n; i += 6) {
        if (n % i == 0 || n % (i + 2) == 0)
            return 0;
    }

    return 1;
}

int primes[10000];
int np; // # of primes
        
void goldbach(int n) {
    int mindiff = 99999;
    int mp1, mp2, i;
    for (i=0; primes[i] < n; i++) {
        if (isPrime(n - primes[i])) {
            int p1 = primes[i], p2 = n - primes[i]; 
            // goldbach partition
            int diff = p1 - p2;
            diff = diff > 0 ? diff : -diff;
            if (mindiff > diff) {
                mindiff = diff;
                mp1 = p1;
                mp2 = p2;
            }
        }
    }
    printf("%d %d\n", mp1 > mp2 ? mp2 : mp1, mp1 > mp2 ? mp1 : mp2);
}

int main() {
    int t, n;

    // pre-populate primes up to 10000
    for (int i=2; i<=10000; i++) {
        if (isPrime(i)) {
            primes[np++] = i;
        }
    }

    scanf("%d", &t);
    while (t--) {
        scanf("%d", &n);
        goldbach(n);
    }
    return 0;
}
```
