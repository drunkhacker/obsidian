한쪽 끝에서 동시 시작하는 투포인터. 부분합(psum)을 구할때 영역 내 (st ~ en) 값을 다 더할필요 없이, 포인터가 가감될때마다 조정하면 된다.

```c
#include <stdio.h>

int a[100000];

int main() {
    int n, s;
    scanf("%d %d", &n, &s);

    for (int i=0; i<n; i++) {
        scanf("%d", &a[i]);
    }

    int st = 0, en = 0, len = 0, min = n+1, psum;
    psum = a[0];

    while (st <= en && en < n) {
        if (psum < s)  {
            en++;
            psum += a[en];
        } else if (psum >= s) {
            len = en - st + 1;
            if (len < min) {
                min = len;
            }
            psum -= a[st];
            st++;
        }
    }

    printf("%d\n", min > n ? 0 : min);

    return 0;
}
```
