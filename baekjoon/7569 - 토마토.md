[[7576 - 토마토]]와 비슷한데 3차원... 크게 다를건 없음.

```
#include <stdio.h>

#define N 100
int map[N][N][N];
int visited[N][N][N];

typedef struct {
    int x;
    int y;
    int z;
} Point;

Point q[N*N*N];
int qhead;
int qtail;

int m,n,h,c;


#define LOG 0
#define write_log(format,args...)        \
                  if (LOG) {                      \
                      printf(format, ## args);    \
                  }


void check_mark(Point u, int x, int y, int z) {
    if (!(x >= 0 && x < m && y >= 0 && y < n && z >= 0 && z < h && map[z][y][x] >= 0)) {
        return;
    }

    if (visited[z][y][x] == 0) {
        visited[z][y][x] = visited[u.z][u.y][u.x] + 1;
        q[qtail++] = (Point){x,y,z};
        write_log("visit (%d,%d,%d)=%d\n",x,y,z,visited[z][y][x]);
    } else if (visited[u.z][u.y][u.x] + 1 < visited[z][y][x]) {
        visited[z][y][x] = visited[u.z][u.y][u.x] + 1;
        q[qtail++] = (Point){x,y,z};
        write_log("update (%d,%d,%d)=%d\n",x,y,z,visited[z][y][x]);
    }
}

int main() {
    int i,j,k;

    scanf("%d%d%d", &m, &n ,&h);
    for (k=0; k<h; k++){ 
        for (i=0; i<n; i++) {
            for (j=0; j<m; j++) {
                scanf("%d", &map[k][i][j]);

                if (map[k][i][j] == 1) {
                    visited[k][i][j] = 1;
                    q[qtail++] = (Point){.x=j, .y=i, .z=k};
                }
            }
        }
    }

    while (qhead != qtail) {
        Point u = q[qhead++];
        write_log("qhead=%d, (%d,%d,%d)\n", qhead, u.x, u.y, u.z);
        int x,y,z;

        x = u.x-1; y=u.y; z=u.z;
        check_mark(u,x,y,z);
        x = u.x+1; y=u.y; z=u.z;
        check_mark(u,x,y,z);
        x = u.x; y=u.y-1; z=u.z;
        check_mark(u,x,y,z);
        x = u.x; y=u.y+1; z=u.z;
        check_mark(u,x,y,z);
        x = u.x; y=u.y; z=u.z-1;
        check_mark(u,x,y,z);
        x = u.x; y=u.y; z=u.z+1;
        check_mark(u,x,y,z);
    }


    // check all covered
    c = 0;
    for (k=0; k<h; k++) { 
        for (i=0; i<n; i++) {
            for (j=0; j<m; j++) {
                if (visited[k][i][j] == 0 && map[k][i][j] != -1) {
                    printf("-1\n");
                    return 0;
                }

                if (visited[k][i][j] > c) {
                    c = visited[k][i][j];
                }
            }
        }
    }

    printf("%d\n", c-1);
    return 0;
}
```
