단순한 dfs. 사이즈가 작아서 adj array로 풀어도 될듯.

```c
#include <stdio.h>

#define N 101

int adj[N][N];
int visited[N];

int c, n;
void dfs(int r) {
    int i;
    visited[r] = 1;
    c++;

    for (i=1; i<=n; i++) {
        if (r != i && adj[r][i] && !visited[i]) {
            dfs(i);
        }
    }
}

int main() {
    int u,v,t;
    scanf("%d%d", &n, &t);
    while (t--) {
        scanf("%d%d", &u, &v);
        adj[u][v] = 1;
        adj[v][u] = 1;
    }

    dfs(1);

    printf("%d\n", c-1);
    return 0;
}

```
