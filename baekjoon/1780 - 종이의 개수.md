9개로 쪼갠다. 아주 작은 트릭은 count 어레이에 인덱싱할때 a\[x\]\[y\] 값을 이용해서 인덱싱 (어차피 -1, 0,1 이므로)


```
#include <stdio.h>

int a[2187][2187] = {0,};
int c[3] = {0,};

void count(int x1, int y1, int x2, int y2, int sz) {
    int x,y,f;
    f = 1;

    for (x=x1; x<x2 && f; x++) {
        for (y=y1; y<y2 && f; y++) {
            if (a[x][y] != a[x1][y1]) {
                f = 0;
            }
        }
    }

    if (f) {
        c[a[x1][y1]+1]++;
    } else {
        for (x=x1; x<x2; x += sz/3) {
            for (y=y1; y<y2; y += sz/3) {
                count(x, y, x+sz/3, y+sz/3, sz/3);
            }
        }
    }
}

int main() {
    int n,x,y;

    scanf("%d", &n);
    for (y=0; y<n; y++) {
        for (x=0; x<n; x++) {
            scanf("%d", &a[x][y]);
        }
    }

    count(0, 0, n, n, n);
    printf("%d\n%d\n%d\n", c[0],c[1],c[2]);

    return 0;
}
```
