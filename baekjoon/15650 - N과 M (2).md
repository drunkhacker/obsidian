[[15649 - N과 M (1)]]와 비슷한데 오름차순 수열만 받는다..
이러면 그냥 visit 배열도 없어도 됨. 간단하게 하려면 마지막에 들렀던 index를 주고 그 뒤로부터만 찾게 하면 되기야하겠다.

```
#include <stdio.h>

int a[] = {0,0,0,0,0,0,0,0,0};
int v[] = {0,0,0,0,0,0,0,0,0};

void f(int n, int m, int d, int j) {
    int i;
    if (m == 0) {
        for (i=0; i<d; i++) {
            printf("%d ", a[i]);
        }
        printf("\n");
        return;
    }
    for (i=j+1; i<=n; i++) {
        if (!v[i]) {
            v[i] = 1;
            a[d] = i;
            f(n, m-1, d+1, i);
            v[i] = 0;
        }
    }
}

int main() {
    int n, m;

    scanf("%d %d", &n, &m);
    f(n, m, 0, 0);
}

```
