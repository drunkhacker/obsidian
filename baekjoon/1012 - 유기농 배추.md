[[2667 - 단지번호붙이기]] 비슷하다.

```c
#include <stdio.h>

#define N 50

int map[N][N];
int visited[N][N];

int m, n, c;

void dfs(int y, int x) {
    visited[y][x] = 1;

    if (x > 0 && map[y][x-1] && !visited[y][x-1]) {
        dfs(y, x-1);
    } 
    if (x < m-1 && map[y][x+1] && !visited[y][x+1]) {
        dfs(y, x+1);
    } 
    if (y > 0 && map[y-1][x] && !visited[y-1][x]) {
        dfs(y-1, x);
    } 
    if (y < n-1 && map[y+1][x] && !visited[y+1][x]) {
        dfs(y+1, x);
    } 
}

int main() {
    int t, x, y, k;
    scanf("%d", &t);

    while (t--) {
        c = 0;
        scanf("%d%d%d", &m, &n, &k);
        while (k--) {
            scanf("%d%d", &x, &y);
            map[y][x] = 1;
        }


        for (y=0; y<n; y++) {
            for (x=0; x<m; x++) {
                if (map[y][x] && !visited[y][x]) {
                    dfs(y, x);
                    c++;
                }
            }
        }
        printf("%d\n", c);

        // teardown
        for (y=0; y<n; y++) {
            for (x=0; x<m; x++) {
                map[y][x] = 0;
                visited[y][x] = 0;
            }
        }
    }

    return 0;
}
```
