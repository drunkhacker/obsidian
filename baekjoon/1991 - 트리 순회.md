https://www.acmicpc.net/problem/1991

입력을 받으면서 트리를 바로 구성하려하면 너무 힘들고, 입력 단계와 트리 구성 단계를 분리하자.
입력은 많아야 26줄이다. `lines[x][0],[1]`이 각각 x가 parent인 트리의 왼쪽, 오른쪽 자식을 나타낸다고 세팅하였다. `build`함수를 재귀적으로 호출하면서 트리를 구성한다. 현재 노드의 data를 이용해 lines 배열을 조회하고 왼쪽, 오른쪽 자식이 있으면 노드를 생성하여 붙여준 후 재귀적으로 내려가면서 호출.

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct tree Tree;
typedef struct tree {
    char data;
    Tree *left;
    Tree *right;
} Tree;

Tree *newnode(char data) {
    Tree *t = malloc(sizeof(Tree));
    t->data = data;
    t->left = NULL;
    t->right = NULL;
    return t;
}

void preorder(Tree *t) {
    if (t == NULL) {
        return;
    }
    putchar(t->data);
    preorder(t->left);
    preorder(t->right);
}

void inorder(Tree *t) {
    if (t == NULL) {
        return;
    }
    inorder(t->left);
    putchar(t->data);
    inorder(t->right);
}

void postorder(Tree *t) {
    if (t == NULL) {
        return;
    }
    postorder(t->left);
    postorder(t->right);
    putchar(t->data);
}

char lines[26][2];

void build(Tree *t) {
    char *line = lines[t->data - 'A'];
    if (line[0] != '.') {
        Tree *l = newnode(line[0]);
        t->left = l;
        build(t->left);
    }
    if (line[1] != '.') {
        Tree *r = newnode(line[1]);
        t->right = r;
        build(t->right);
    }
}

int main() {
    int n;
    scanf("%d", &n);

    char p, l, r;
    // input
    for (int i=0; i<n; i++) {
        while ((getchar()) != '\n');
        scanf("%c %c %c", &p, &l, &r);
        lines[p-'A'][0] = l;
        lines[p-'A'][1] = r;
    }

    // make tree (link nodes)
    Tree *root = newnode('A');
    build(root);

    // traversal
    // preorder
    preorder(root);
    putchar(10);
    // inorder
    inorder(root);
    putchar(10);
    // postorder
    postorder(root);
    putchar(10);

    return 0;
}

```
