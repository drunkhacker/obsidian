다만 초기조건이 front = -1인건 왜있는진 모르겠음.

```
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define N 10001

struct dequeue {
    int arr[N];
    int front, rear;
} deq;

void init(struct dequeue* deq) {
    deq->front = -1;
    deq->rear = 0;
}

int is_empty(struct dequeue* deq) {
    return deq->front == -1;
}

void push_front(struct dequeue* deq, int x) {
    if (is_empty(deq)) { // initial empty
        deq->front = 0;
        deq->rear = 0;
    } else if (deq->front == 0) {
        deq->front = N-1;
    } else {
        deq->front--;
    }

    deq->arr[deq->front] = x;
}

void push_back(struct dequeue* deq, int x) {
    if (is_empty(deq)) {
        deq->front = 0;
        deq->rear = 0;
    } else if (deq->rear == N-1) {
        deq->rear = 0;
    } else {
        deq->rear++;
    }

    deq->arr[deq->rear] = x;
}

int size(struct dequeue* deq) {
    int s;

    if (is_empty(deq)) {
        return 0;
    } 

    s = deq->rear - deq->front + 1;

    return s > 0 ? s : N + s;
}

int pop_front(struct dequeue* deq) {
    int r;

    if (is_empty(deq)) {
        return -1;
    } 

    r = deq->arr[deq->front];
    if (deq->front == deq->rear) {
        deq->front = -1;
        deq->rear = -1;
    } else if (deq->front == N-1) {
        deq->front = 0;
    } else {
        deq->front++;
    }

    return r;
}

int pop_back(struct dequeue* deq) {
    int r;

    if (is_empty(deq)) {
        return -1;
    } 

    r = deq->arr[deq->rear];
    if (deq->front == deq->rear) {
        deq->front = -1;
        deq->rear = -1;
    } else if (deq->rear == 0) {
        deq->rear = N-1;
    } else {
        deq->rear--;
    }

    return r;
}

int front(struct dequeue* deq) {
    if (is_empty(deq)) {
        return -1;
    }

    return deq->arr[deq->front];
}

int back(struct dequeue* deq) {
    if (is_empty(deq)) {
        return -1;
    }

    return deq->arr[deq->rear];
}

int main() {
    int n, i, x;
    char *pc;
    char cmd[30];

    scanf("%d", &n);
    while ((getchar()) != '\n');

    init(&deq);
    for (i=0; i<n; i++) {
        fgets(cmd, 29, stdin);
        if (strncmp(cmd, "push_back", 9) == 0) {
            pc = cmd;
            strsep(&pc, " ");
            x = atoi(strsep(&pc, " "));
            push_back(&deq, x);
        } else if (strncmp(cmd, "push_front", 10) == 0) {
            pc = cmd;
            strsep(&pc, " ");
            x = atoi(strsep(&pc, " "));
            push_front(&deq, x);
        } else if (strncmp(cmd, "size", 4) == 0) {
            printf("%d\n", size(&deq));
        } else if (strncmp(cmd, "empty", 5) == 0) {
            printf("%d\n", is_empty(&deq));
        } else if (strncmp(cmd, "front", 5) == 0) {
            printf("%d\n", front(&deq));
        } else if (strncmp(cmd, "back", 4) == 0) {
            printf("%d\n", back(&deq));
        } else if (strncmp(cmd, "pop_front", 9) == 0) {
            printf("%d\n", pop_front(&deq));
        } else if (strncmp(cmd, "pop_back", 8) == 0) {
            printf("%d\n", pop_back(&deq));
        }  else {
            printf("cannot parse cmd=%s\n", cmd);
        }
    }

    return 0;
}
```
