bfs로 풀 수 있음.
순간이동할때와 그냥 움직일때 visit 거리 업데이트 값이 달라지는것을 check_mark 함수의 인자로 넘기자.

```c
#include <stdio.h>

#define N 100001

int visit[N];
#define NQ (3*N)
int q[NQ];
int qtail, qhead;

void check_mark(int u, int x, int d) {
    if (x >= 0 && x < N && (visit[x] == 0 || visit[x] > visit[u]+d)) {
        visit[x] = visit[u] + d;
        q[qtail] = x;
        qtail = (qtail+1)%NQ;
    }
}

int main() {
    int n,k;
    scanf("%d%d", &n, &k);

    visit[n] = 1;

    q[qtail] = n;
    qtail = (qtail+1)%NQ;

    while (qhead != qtail) {
        int u = q[qhead];
        qhead = (qhead+1)%NQ;

        int x;
        x = u-1;
        check_mark(u,x,1);

        x = u+1;
        check_mark(u,x,1);

        x = 2*u;
        check_mark(u,x,0);
    }

    printf("%d\n", visit[k] - 1);

    return 0;
}
```
