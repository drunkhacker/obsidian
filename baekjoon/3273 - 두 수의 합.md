https://www.acmicpc.net/problem/3273

소팅 후 two pointer. 양 끝에서 출발하자.

```c
#include <stdio.h>
#include <stdlib.h>

int comp(const void *a, const void *b) {
    return (*(int *)a) - (*(int *)b);
}

int a[100000];

int main() {
    int i, n, x;
    scanf("%d", &n);

    for (i=0; i<n; i++) {
        scanf("%d", &a[i]);
    }

    scanf("%d", &x);

    // sort
    qsort(a, n, sizeof(int), comp);

    int s, e, c;
    s = 0; e = n-1;
    c = 0;

    while (s < e) {
        if (a[s] + a[e] == x) {
            s++; c++;
        } else if (a[s] + a[e] < x) {
            s++;
        } else {
            e--;
        }
    }

    printf("%d\n", c);

    return 0;
}

```
