정렬 후, 양 끝에서 투포인터.

```c
#include <stdio.h>
#include <stdlib.h>

#define abs(x) ((x) >= 0 ? (x) : (-(x)))

int comp(const void *a, const void *b) {
    return (*(int *)a) - (*(int *)b);
}

int a[100000];

int main() {
    int n, i;

    scanf("%d", &n);
    for (i=0; i<n; i++) {
        scanf("%d", &a[i]);
    }

    qsort(a, n, sizeof(int), comp);

    int s, e, ss, ee, min, sum;

    min = 2100000000;
    s = ss = 0; e = ee = n-1; 

    while (s < e) {
        sum = abs(a[s] + a[e]);
        if (sum < min) {
            min = sum;
            ss = s; ee = e;
        }

        if (sum == 0) {
            break;
        } else if (a[s] + a[e] > 0) {
            e--;
        } else if (a[s] + a[e] < 0) {
            s++;
        }

    }

    printf("%d %d\n", a[ss], a[ee]);

    return 0;
}
```
