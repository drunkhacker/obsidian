https://www.acmicpc.net/problem/1504

v1을 먼저 들르는게 더 경제적일수도 있고,  v2를 먼저 들르는게 더 경제적일수도 있어서 두 경우 다 해보는게 필요하다. 
즉, 
a) 1 -> v1 , v1 -> v2, v2 -> N
b) 1 -> v2, v2 -> v1, v1 -> N
두 경우 중 더 경제적인것이 답.

몇번의 시도에서 계속 out of bounds 에러가 났었는데, priority q에 들어갈 수 있는 원소가 간선만큼 들어갈 수 있다는것을 생각하지 않았었음. (코드에서 (1) 부분)

```c
#include <stdio.h>
#include <stdlib.h>

#define MAX_V 800
#define MAX_E 200000
#define INF 1e9

#define LOG 0
#define write_log(format,args...)        \
                  if (LOG) {                      \
                      printf(format, ## args);    \
                  }

// priority queue

typedef struct {
    int prio;
    int elem;
} QElem;

struct {
    QElem *arr[MAX_E]; // <----- (1)
    int size;
} q;

void percolateDown(int i) {
    int li = 2*i+1, ri = 2*i+2;

    if (li >= q.size) {
        return;
    }

    int target;
    if (ri >= q.size || q.arr[li]->prio < q.arr[ri]->prio) {
        target = li;
    } else {
        target = ri;
    }

    if (q.arr[target]->prio < q.arr[i]->prio) {
        QElem *t = q.arr[i];
        q.arr[i] = q.arr[target];
        q.arr[target] = t;
        percolateDown(target);
    }
}

void percolateUp(int i) {
    if (i == 0) {
        return;
    }
    int pi = (i-1)/2;
    // check is heap
    if (q.arr[pi]->prio > q.arr[i]->prio) {
        QElem *t = q.arr[i];
        q.arr[i] = q.arr[pi];
        q.arr[pi] = t;
        percolateUp(pi);
    }
}

QElem* pop() {
    if (q.size <= 0) {
        return NULL;
    }
    QElem *t = q.arr[0];
    q.arr[0] = q.arr[q.size-1];
    q.arr[q.size-1] = NULL;
    q.size--;
    percolateDown(0);
    return t;
}

void insert(QElem *x) {
    // insert to bin tree
    q.arr[q.size] = x;
    percolateUp(q.size++);
}

int g[MAX_V+1][MAX_V+1];
int dist[MAX_V+1];
int N;

int dijkstra(int src, int dest) {
    if (src == dest) {
        return 0;
    }

    for (int v=1; v<=N; v++) {
        dist[v] = INF;
    }

    dist[src] = 0;
    QElem *e = malloc(sizeof(QElem));
    e->elem = src;
    e->prio = 0;
    insert(e);

    while (q.size > 0) {
        QElem *u = pop();

        if (dist[u->elem] < u->prio) {
            continue;
        }

        for (int v=1; v<=N; v++) {
            int w;
            if ((w = g[u->elem][v]) == 0) { // no edge between u-v
                continue;
            }
            if (dist[v] > dist[u->elem] + w) {
                dist[v] = dist[u->elem] + w;

                write_log("update dist to %d to %d\n", v, dist[v]);

                e = malloc(sizeof(QElem));
                e->elem = v;
                e->prio = dist[v];
                insert(e);
            }
        }
        free(u);
    }

    return dist[dest];
}


int main() {
    int E, v1, v2;
    scanf("%d%d", &N, &E);

    while (E) {
        int a, b, c;
        scanf("%d%d%d", &a,&b,&c);
        g[a][b] = g[b][a] = c;
        E--;
    }

    scanf("%d%d", &v1,&v2);
    long d1 = (long)dijkstra(1, v1) + dijkstra(v1, v2) + dijkstra(v2, N);
    long d2 = (long)dijkstra(1, v2) + dijkstra(v2, v1) + dijkstra(v1, N);
    long d = d1 > d2 ? d2 : d1;

    printf("%ld\n", d >= INF ? -1 : d);
    return 0;
}
```
