간단한 소팅 후 이분탐색인데, 소팅할때 comp 함수.. 에서 언더플로 날 수 있음. 그냥 a- b 로 comp를 만들면 안됨!


```
#include <stdio.h>
#include <stdlib.h>

int a[100000];

int comp(const void *a, const void *b) {
    return (*(int *)a) > (*(int *)b) ? 1 : (*(int *)a) < (*(int *)b) ? -1 : 0;
}

int main() {
    int n, m, i, x, st, en, mid, f;
    scanf("%d", &n);
    for (i=0; i<n; i++) {
        scanf("%d", &a[i]);
    }

    qsort(a, n, sizeof(int), comp);

    scanf("%d", &m);
    for (i=0; i<m; i++) {
        scanf("%d", &x);

        f = 0;
        st = 0; en = n-1;
        while (st <= en && !f) {
            mid = (st + en) /2;
            if (a[mid] == x) {
                f = 1;
                break;
            } else if (a[mid] > x) {
                en = mid - 1;
            } else {
                st = mid + 1;
            }
        }
        printf("%d\n", f);
    }
    return 0;
}
```
