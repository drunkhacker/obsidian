특이한건 없고 다만 `bt[]`배열에 기록하는 element의 기본값이 char '0' 이고 이걸 조건문에서 제대로 거르지 못할 경우, 화면에는 나오지 않지만 입출력 파일 상으로는 다르게 나올 수 있다. 

그리고 꼭 check_mark상에서 relaxation이 일어날때를 고민하지 않아도 된다. 어차피 BFS는 최단경로를 찾게 돼있음.

```c
#include <stdio.h>

#define LOG 0
#define write_log(format,args...)              \
                  if (LOG) {                   \
                      printf(format, ## args); \
                  }


#define N 10000
typedef struct { 
    int x;
    char op;
} elem;

int visit[N];
elem bt[N];

#define NQ (10*N)
int q[NQ];
int qtail, qhead;

void check_mark(int u, int x, char op) {
    if (!visit[x]) {
        visit[x] = 1;
        q[qtail] = x;
        qtail = (qtail+1)%NQ;

        // u -> x
        write_log("bt[%d] = (%d, %c)\n", x, u, op);
        bt[x] = (elem){u, op};
    }
}


void bfs(int a, int b) {
    int i;
    for (i=0; i<N; i++) {
        visit[i] = 0;
        bt[i] = (elem){0, '?'};
    }
    for (i=0; i<NQ; i++) {
        q[i] = 0;
    }

    qhead = qtail = 0;

    visit[a] = 1;
    q[qtail] = a;
    qtail = (qtail + 1)%NQ;

    while (qhead != qtail) {
        int u = q[qhead];
        if (u == b) {
            break;
        }
        qhead = (qhead+1)%NQ;

        int x;
        // 2n
        x = (u*2)%10000;
        check_mark(u, x, 'D');

        // n-1
        x = u == 0 ? 9999 : u-1;
        check_mark(u, x, 'S');

        // left
        int d1 = u/1000;
        x = (u%1000)*10 + d1;
        check_mark(u, x, 'L');

        // right
        int d4 = u%10;
        x = d4*1000 + u/10;
        check_mark(u, x, 'R');
    }
}

void print_back(int b, int a) {
    write_log("print_back a=%d, b=%d\n", a, b);
    if (a != b) {
        print_back(bt[b].x, a);
        write_log("bt[%d].op=%c\n", b, bt[b].op);
        printf("%c", bt[b].op);
    }
} 

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        int a, b;
        scanf("%d%d", &a, &b);
        write_log("call bfs func\n");
        bfs(a, b);
        write_log("call print_back func\n");
        print_back(b, a);
        printf("\n");
    }
    return 0;
}

```
