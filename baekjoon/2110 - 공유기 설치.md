특정 거리 m 이 조건을 만족하느냐를 체크할때,
좌표를 정렬해놓고 좌표사이 거리를 재가면서 그 거리가 >= m 일때 cnt ++ 해보는 방식으로.  cnt가 공유기 갯수 -1 이상 나온다면 최인접 거리는 m보다 더 길어질 수 있음.

```
#include <stdio.h>
#include <stdlib.h>

int a[200000];
int n, c;

int solve() {
    int st, en, i, j, m, cnt;

    st = a[n-1] - a[0];
    en = a[n-1] - a[0];
    for (i=0; i<n-1; i++) {
        if (a[i+1] - a[i] < st) {
            st = a[i+1] - a[i];
        }
    }

    while (st < en) {
        m = ((long)st + en + 1)/2;

        // check m is possible..
        for (cnt=0, i=0; i<n-1;) {
            for (j=i+1; j<n; j++) {
                if (a[j] - a[i] >= m) {
                    cnt++;
                    i = j; 
                    break;
                }
            }

            if (j == n) {
                break;
            }
        }

        if (cnt >= c-1) {
            st = m;
        } else {
            en = m - 1;
        }
            
    }

    return st;
}

int cmp(const void *a, const void *b) {
    return (*(int *)a) > (*(int *)b) ? 1 : (*(int *)a) < (*(int *)b) ? -1 : 0;
}

int main() {
    int i;

    scanf("%d %d", &n, &c);
    for (i=0; i<n; i++) {
        scanf("%d", &a[i]);
    }

    qsort(a, n, sizeof(int), cmp);

    printf("%d\n", solve());

    return 0;
}
```
