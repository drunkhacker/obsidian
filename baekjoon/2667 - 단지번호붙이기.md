2차원 map표현에서 인접한 노드를 찾는것 + dfs

```
#include <stdio.h>

#define N 25
int map[N][N];
int visited[N*N];
int pp[N*N];

int n, c, t;

void dfs(int y, int x) {
    visited[x+y*n] = 1;
    c++;

    if (x > 0 && map[y][x-1] && !visited[(x-1) + y*n]) {
        dfs(y, x-1);
    } 
    if (x < n-1 && map[y][x+1] && !visited[(x+1) + y*n]) {
        dfs(y, x+1);
    } 
    if (y > 0 && map[y-1][x] && !visited[x + (y-1)*n]) {
        dfs(y-1, x);
    } 
    if (y < n-1 && map[y+1][x] && !visited[x + (y+1)*n]) {
        dfs(y+1, x);
    } 
}


int main() {
    int i,j;

    scanf("%d", &n);
    while ((getchar()) != '\n');

    for (i=0; i<n; i++) {
        for (j=0; j<n; j++) {
            map[i][j] = fgetc(stdin) - '0';
        }
        while ((getchar()) != '\n');
    }

    for (i=0; i<n; i++) {
        for (j=0; j<n; j++) {
            c = 0;
            if (map[i][j] && !visited[i*n+j]) {
                dfs(i, j);
                pp[t++] = c;
            }
        }
    }

    for (i=0; i<t-1; i++) {
        for (j=0; j<t-i-1; j++) {
            if (pp[j] > pp[j+1]) {
                int x = pp[j];
                pp[j] = pp[j+1];
                pp[j+1] = x;
            }
        }
    }

    printf("%d\n", t);
    for (i=0; i<t; i++) {
        printf("%d\n", pp[i]);
    }

    return 0;
}
```
