각 원소당 인덱스를 두종류 (왼쪽부터, 오른쪽부터) 매긴다고 생각해보자.
정확히는 몇번 더 회전연산을 해야 해당 원소가 왼쪽 첫번째에 오는가? 에 대한 수치를 저장하고 있다고 생각.
각 인덱스를 i1, i2 라고 하면
i2 = N - i1 (N은 현재 배열의 원소 갯수다)

- 왼쪽으로 돌릴때는 인덱스의 첫번째 인덱스를 -1 하면서 두번째 인덱스는 +1
- 오른쪽으로 돌릴때는 인덱스의 두번째 인덱스를 -1 하면서 첫번째 인덱스는 +1
- 0이 떠서 뺄때는 왼쪽 인덱스만 -1 

답= 1+3+4 = 8

#include <stdio.h>

int indexes[51][2] = {0,};

void print_index(int s, int m, int n) {
    int i;
    printf("(");
    for (i=s; i<m; i++) {
        printf("(%d,%d), ", indexes[i][0], indexes[i][1]);
    }
    printf(")\n");
}

int main() {
    int n, m, i, x, c, a;

    scanf("%d %d", &n, &m);
    for (i=0; i<m; i++) {
        scanf("%d", &x);
        indexes[i][0] = x-1;
        indexes[i][1] = n - (x-1);
    }


    c = 0;
    a = 0;

    // print_index(c, m, n);
    while (c < m) {
        if (indexes[c][0] == 0 || indexes[c][1] == 0) { // remove first
            c++; n--;
            for (i=c; i<m; i++) {
                indexes[i][0] -= 1;
            }
            // printf("pop the first one, c = %d\n", c);
            // print_index(c, m, n);
        } else {
            if (indexes[c][0] <= indexes[c][1])  { // left rotate is smaller
                x = indexes[c][0];
                a += x;
                // printf("rotate left %d times\n", x);
                for (i=c; i<m; i++) {
                 //   printf("i=%d, indexes[c]=%d\n", i, x);
                    indexes[i][0] -= x;
                    if (indexes[i][0] < 0) {
                        indexes[i][0] += n;
                    }
                    indexes[i][1] = (indexes[i][1] + x) % n;
                }
            } else { // right
                x = indexes[c][1];
                a += x;
                // printf("rotate right %d times\n", x);
                for (i=c; i<m; i++) {
                    indexes[i][1] -= x;
                    if (indexes[i][1] < 0) {
                        indexes[i][1] += n;
                    }
                  //  printf("i=%d\n", i);
                    indexes[i][0] = (indexes[i][0] + x) % n;
                }
            }
            // print_index(c, m, n);
        }
    }

    printf("%d\n", a);
    return 0;
}
