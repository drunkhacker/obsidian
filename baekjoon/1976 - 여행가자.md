[[1717 - 집합의 표현]]과 같이 union find 로 풀 수 있다.
노드간의 연결을 합집합(union)연산으로 보자. 그리고 마지막 여행 path가 가능한지의 여부는 path의 모든 노드들이 다 같은 집합에 속하는지를 확인하면 된다.

```c
#include <stdio.h>

#define N 200

int parent[N+1];

int find(int x) {
    if (parent[x] == x) {
        return x;
    }

    int r = find(parent[x]);
    parent[x] = r;
    return r;
}

void merge(int a, int b) {
    int ra = find(a);
    int rb = find(b);

    parent[rb] = ra;
}

int main() {
    int i, j, n, m, x, r, f;
    for (i=0; i<=N; i++) {
        parent[i] = i;
    }
    scanf("%d %d", &n, &m);
    for (i=1; i<=n; i++) {
        for (j=1; j<=n; j++) {
            scanf("%d", &x);
            if (x == 1) {
                merge(i, j);
            }
        }
    }

    r = 1;
    scanf("%d", &x);
    f = find(x);
    for (i=1; i<m && r; i++) {
        scanf("%d", &x);
        r = r && f == find(x);
    }
    printf("%s\n", r ? "YES" : "NO");
    return 0;
}

```
