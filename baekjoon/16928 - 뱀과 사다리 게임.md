BFS 최단경로 문제.
jump 배열에 사다리정보나 뱀의 정보를 담고, 사다리나 뱀을 타서 다른 칸으로 워프할때와 그냥 주사위를 굴릴때의 경우를 나눠서 한다.

```
#include <stdio.h>

int jump[101];
int visit[101];

int q[60000];
int qhead;
int qtail;

void check_mark(int u, int x, int d) {
    if (x >= 1 && x <= 100 && (visit[x] == 0 || (visit[x] > visit[u]+d))) {
        visit[x] = visit[u] + d; 
        q[qtail++] = x;
    }
}

int main() {
    int n,m,i,x;
    scanf("%d%d", &n, &m);

    for (i=0; i<n+m; i++) {
        scanf("%d", &x);
        scanf("%d", &jump[x]);
    }

    visit[1] = 1;
    q[qtail++] = 1;

    while (qhead != qtail) {
        int u = q[qhead++];

        if (jump[u]) {
            check_mark(u, jump[u], 0);
        } else {
            for (i=1;i<=6;i++) {
                check_mark(u, u+i, 1);
            }
        }
    }

    printf("%d\n", visit[100] - 1);

    return 0;
}
```
