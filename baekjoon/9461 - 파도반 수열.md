p(n) = p(n-1) + p(n-5)인데, 계산하다보면 int 범위 넘어간다.

```
#include <stdio.h>

long p[101] = {0,1,1,1,2,2,3,4,5,7,9,0,};
int main() {
    int t, n, i;

    scanf("%d", &t);
    while (t-- > 0) {
        scanf("%d", &n);
        if (p[n] > 0) {
            printf("%ld\n", p[n]);
            continue;
        }
        for (i = 10; i <= n; i++) {
            p[i] = p[i-5] + p[i-1];
        }
        printf("%ld\n", p[n]);
    }

    return 0;
}

```
