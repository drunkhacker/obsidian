스택에 괄호 넣기. 근데 스택도 필요없고 counter같은거 둬서 마이너스로 빠지는 순간 튀면 됨

```
#include <stdio.h>
char str[BUFSIZ];

int main() {
    int n, i, c;
    char *pc;
    scanf("%d", &n);
    for (i=0; i<n; i++) {
        scanf("%s", str);
        c = 0;
        pc = str;
        while (*pc) {
            c += (*pc == '(') ? 1:-1;
            if (c < 0) {
                break;
            }
            pc++;
        }
        printf("%s\n", c == 0 ? "YES" : "NO");
    }
    return 0;
}

```
