[[24479 - 알고리즘 수업 - 깊이 우선 탐색 1]]와 똑같은데 방향만 다른 것.

```
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define N 100000

#define LOG 0
#define write_log(format,args...)                 \
                  if (LOG) {                      \
                      printf(format, ## args);    \
                  }

int visited[N+1];
int order = 1;

typedef struct node Node;
struct node {
    int n;
    Node* next;
};

typedef struct graph Graph;
struct graph {
    int size; // # of nodes
    Node *adj[N+1];
    Node *tails[N+1];
} g;

void print_graph() {
#if LOG
    for (int i=1; i<=g.size; i++) {
        if (g.adj[i]) {
            printf("adj[%d] = ", i);
            Node *n = g.adj[i];
            while (n) {
                printf("%d, ", n->n);
                n = n->next;
            }
            printf("\n");
        }
    }
#endif
}

Node node_pool[N*5];
int pc;

Node *get_new_node() {
    return &node_pool[pc++];
}

void add_edge(int u, int v) {
    Node *node = get_new_node();
    node->next = NULL;
    node->n = v;

    if (!g.adj[u]) {
        g.adj[u] = node;
    } else {
        g.tails[u]->next = node;
    }
    g.tails[u] = node;
}

void split(Node *source, Node **frontRef, Node **backRef) {
    Node *fast;
    Node *slow;

    slow = source;
    fast = source->next;

    while (fast != NULL) {
        fast = fast->next;
        if (fast != NULL) {
            slow = slow->next;
            fast = fast->next;
        }
    }

    *frontRef = source;
    *backRef = slow->next;
    slow->next = NULL;
}

Node *merge(Node *a, Node *b) {
    Node *result = NULL;
    if (a == NULL) { return b; }
    if (b == NULL) { return a; }

    if (a->n >= b->n) {
        result = a;
        result->next = merge(a->next, b);
    } else {
        result = b;
        result->next = merge(a, b->next);
    }

    return result;
}

void mergeSort(Node **headRef) {
    Node *head = *headRef;
    Node *a;
    Node *b;

    if (head == NULL || head->next == NULL) {
        return;
    }

    split(head, &a, &b);
    mergeSort(&a);
    mergeSort(&b);

    *headRef = merge(a, b);
}

void dfs(int r) {
    visited[r] = order++;
    write_log("visited %d\n", r);
    Node *node = g.adj[r];

    if (node) {
        Node *adjlist = node;
        mergeSort(&adjlist);
        g.adj[r] = adjlist;
        node = g.adj[r];
    }

    while (node != NULL) {
        if (visited[node->n] == 0) {
            dfs(node->n);
        }
        node = node->next;
    }
}

int main() {
    int n, m, r, i;
    scanf("%d%d%d", &n, &m, &r);

    g.size = n;

    for (i=0; i<m; i++) {
        int u,v;
        scanf("%d%d", &u, &v);

        add_edge(u, v);
        add_edge(v, u);
    }

    print_graph();

    dfs(r);

    for (i=1; i<=n; i++) {
        printf("%d\n", visited[i]);
    }

    return 0;
}
```
