```
#include <stdio.h>
#include <string.h>

int N;
int b[9][9] = {0,}; // board
int a[81][2] = {0,}; // coord of unknowns
int v[10] = {0,};

int backtrack(int n) {
    int x, r, c, ok, rr, cc, i, j;
    if (n == N) {
        // check all
        // check row by row
        for (r=0; r<9; r++) {
            for (i=0; i<8; i++) {
                for (j=i+1; j<9; j++) {
                    if (b[r][i] == b[r][j]) {
                        return 0;
                    }
                }
            }
        }

        // check col by col
        for (c=0; c<9; c++) {
            for (i=0; i<8; i++) {
                for (j=i+1; j<9; j++) {
                    if (b[i][c] == b[j][c]) {
                        return 0;
                    }
                }
            }
        }

        // check by neighbor
        for (rr=0; rr<3; rr++) {
            for (cc=0; cc<3; cc++) {
                // clear visit arr
                memset(v, 0, sizeof(int)*10);
                for (r=rr*3; r < (rr+1)*3; r++) {
                    for (c=cc*3; c < (cc+1)*3; c++) {
                        if (v[b[r][c]]) {
                            return 0;
                        } else {
                            v[b[r][c]] = 1;
                        }
                    }
                }
            }
        }

        // print board
        for (r=0; r<9; r++) {
            for (c=0; c<8; c++) {
                printf("%d ", b[r][c]);
            }
            printf("%d\n", b[r][c]);
        }

        return 1;
    }

    for (x=1; x<=9; x++) {
        ok = 1;
        // check row
        for (c=0; c<9; c++) {
            if (b[a[n][0]][c] == x) {
                ok = 0;
                break;
            }
        }

        // check col
        if (ok) {
            for (r=0; r<9; r++) {
                if (b[r][a[n][1]] == x) {
                    ok = 0;
                    break;
                }
            }
        }

        // check neighbor
        if (ok) {
            rr = (a[n][0]/3)*3;
            cc = (a[n][1]/3)*3;

            for (r=rr; (r < rr+3) && ok; r++) {
                for (c=cc; c < cc+3; c++) {
                    if (b[r][c] == x) {
                        ok = 0;
                        break;
                    }
                }
            }
        }

        if (!ok) {
            continue;
        }

        // try this candidate
        b[a[n][0]][a[n][1]] = x;

        if (backtrack(n+1)) {
            return 1;
        }

        // reset
        b[a[n][0]][a[n][1]] = 0;
    }
    return 0;
}

int main() {
    int r, c;

    N = 0;
    for (r=0;r<9;r++) {
        for (c=0;c<9;c++) {
            scanf("%d", &b[r][c]);
            if (b[r][c] == 0) {
                a[N][0] = r;
                a[N][1] = c;
                N++;
            }
        }
    }
    backtrack(0);
}
```
