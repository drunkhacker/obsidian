a,b,c 에 대해서 memoization + for loop으로 올라가면서 build up?

대신 m값이 0이라는게 unvisit을 의미하지 않을 수 있으므로 따로 visit 배열을 두는게 낫겠다.
```
#include <stdio.h>

int m[102][102][102] = {0, };
int v[102][102][102] = {0, };

int f(int a, int b, int c) {
    int r;
    if (v[a + 50][b + 50][c + 50]) {
        return m[a + 50][b + 50][c + 50];
    }
    if (a <= 0 || b <= 0 || c <= 0) {
        m[a + 50][b + 50][c + 50] = 1;
        v[a + 50][b + 50][c + 50] = 1;
        return 1;
    }
    if (a > 20 || b > 20 || c > 20) {
        r = f(20,20,20);
        m[a + 50][b + 50][c + 50] = r;
        v[a + 50][b + 50][c + 50] = 1;
        return r;
    }

    if (a<b && b<c) {
        r = f(a,b,c-1) + f(a,b-1,c-1) - f(a,b-1,c);
        m[a+50][b+50][c+50] = r;
        v[a+50][b+50][c+50] = 1;
        return r;
    }

    r = f(a-1, b, c) + f(a-1, b-1, c) + f(a-1, b, c-1) - f(a-1, b-1, c-1);
    m[a+50][b+50][c+50] = r;
    v[a+50][b+50][c+50] = 1;
    return r;
}

int main() {
    int a,b,c,w;
    while (1) {
        scanf("%d %d %d", &a, &b, &c);
        if (a == -1 && b == -1 && c == -1) {
            break;
        }
        w = f(a,b,c);
        printf("w(%d, %d, %d) = %d\n", a,b,c,w);
    }
    return 0;
}
```
