기본 행렬 곱셈 구현..

```
#include <stdio.h>

int a[100][100];
int b[100][100];
int c[100][100];

int main() {
    int n,m,k,i,j,ii;

    scanf("%d%d", &n, &m);
    for (i=0; i<n; i++) {
        for (j=0; j<m; j++) {
            scanf("%d", &a[i][j]); 
        }
    }

    scanf("%d%d", &m, &k);
    for (i=0; i<m; i++) {
        for (j=0; j<k; j++) {
            scanf("%d", &b[i][j]); 
        }
    }

    // (n x m) x (m x k) -> (n x k) 
    for (i=0; i<n; i++) {
        for (j=0; j<k; j++) {
            c[i][j] = 0;

            for (ii=0; ii<m; ii++) {
                c[i][j] += a[i][ii]*b[ii][j];
            }
        }
    }

    for (i=0; i<n; i++) {
        for (j=0; j<k; j++) {
            printf("%d ", c[i][j]);
        }
        printf("\n");
    }
    
    return 0;
}
```
