https://www.acmicpc.net/problem/1956

플로이드 와샬 알고리즘 수행시 자기자신에 대한 dist를 INF로 놓고 시작하면 된다.

```c
#include <stdio.h>

#define INF 1e9
#define N 400

int g[N+1][N+1];
int dist[N+1][N+1];

int V, E;

void floyd() {
    int i, j, k;

    for (k=1; k<=V; k++) {
        for (i=1; i<=V; i++) {
            for (j=1; j<=V; j++) {
                if (dist[i][j] > dist[i][k] + dist[k][j]) {
                    dist[i][j] = dist[i][k] + dist[k][j];
                }
            }
        }
    }
}

int main() {
    int i, j, u, v, w, min;

    scanf("%d%d", &V, &E);

    for (i=1; i<=V; i++) {
        for (j=1; j<=V; j++) {
            g[i][j] = INF;
            dist[i][j] = INF;
        }
    }

    for (i=0; i<E; i++) {
        scanf("%d%d%d", &u, &v, &w);
        g[u][v] = w;
        dist[u][v] = w;
    }

    floyd();

    min = INF;
    for (i=1; i<=V; i++) {
        if (min > dist[i][i]) {
            min = dist[i][i];
        }
    }

    printf("%d\n", min >= INF ? -1 : min);

    return 0;
}

```
