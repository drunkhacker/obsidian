```
#include <stdio.h>

int a[] = {0,0,0,0,0,0,0,0,0};
int v[] = {0,0,0,0,0,0,0,0,0};

void f(int n, int m, int d) {
    int i;
    if (m == 0) {
        for (i=0; i<d; i++) {
            printf("%d ", a[i]);
        }
        printf("\n");
        return;
    }
    for (i=1; i<=n; i++) {
        if (!v[i]) {
            v[i] = 1;
            a[d] = i;
            f(n, m-1, d+1);
            v[i] = 0;
        }
    }
}

int main() {
    int n, m;

    scanf("%d %d", &n, &m);
    f(n, m, 0);
}
```
