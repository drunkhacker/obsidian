dp\[d\]\[n\]= d로 시작하면서 길이가 n인 계단수의 갯수 

dp\[d\]\[n\] = dp\[d-1\]\[n-1\] + dp\[d+1\]\[n-1\]
흠.. 근데 0이 처음엔 못 오지만 중간에는 올수있자나. 그걸 고민하면?

특수조건:
d == 1이면 
1 2 ... 뿐 아니라
1 0 1 ... 도 고민을 해야함.

dp 돌린 후 sum(dp\[1\]\[n\] ... dp\[9\]\[n\]) 하면 될듯 

for loop을 n에 대해서 밖으로 돌리고 d에 대해서 안에서 돌리면 되나?

```
for (n in (1..N))
  for (d in (1..9))
    if (d == 1):
      dp[d][n] = dp[2][n-1] + dp[1][n-2]
    else:
      dp[d][n] = dp[d-1][n-1] + dp[d+1][n-1] // d+1 <= 9 주의
```

1차시도
```
#include <stdio.h>

int dp[10][101] = {0,};
int c = 1000000000;

int main() {
    int n, i, d, s;
    scanf("%d", &n);

    for (d=1; d<=9; d++) {
        dp[d][1] = 1;
        dp[d][2] = 2;
    }
    dp[9][2] = 1;

    for (i=3; i<=n; i++) {
        for (d=1; d<=9; d++) {
            if (d == 1) {
                dp[1][i] = dp[2][i-1] % c + dp[1][i-2] % c;
            } else if (d < 9) {
                dp[d][i] = dp[d-1][i-1] % c + dp[d+1][i-1] % c;
            } else {
                dp[d][i] = dp[d-1][i-1] % c;
            }
        }
    }

    s = 0;
    for (d=1; d<=9; d++) {
        s = (s + dp[d][n]) % c;
    }

    printf("%d\n", s);

    return 0;
}
```

```
#include <stdio.h>

int dp[10][101] = {0,};
int c = 1000000000;

int main() {
    int n, i, d;
    int s;
    scanf("%d", &n);

    for (d=1; d<=9; d++) {
        dp[d][1] = 1;
        dp[d][2] = 2;
    }
    dp[9][2] = 1;

    for (i=3; i<=n; i++) {
        for (d=1; d<=9; d++) {
            if (d == 1) {
                dp[1][i] = (dp[2][i-1] + dp[1][i-2])%c;
            } else if (d < 9) {
                dp[d][i] = (dp[d-1][i-1] + dp[d+1][i-1])%c;
            } else {
                dp[d][i] = dp[d-1][i-1];
            }
        }
    }

    s = 0;
    for (d=1; d<=9; d++) {
        s = (s + dp[d][n]) % c;
    }

    printf("%d\n", s);

    return 0;
}
```
