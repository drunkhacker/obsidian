플로이드 워셜 알고리즘을 적용한다. 그래프 입력을 받을때 두 지점간 간선이 2개 이상일 수 있는데, 가중치가 더 작은 간선을 택해서 그래프를 구성해야 한다.

```c
#include <stdio.h>
#define INF 1e10
#define N 100

long g[N+1][N+1];
long dist[N+1][N+1];

int n, m;

void floyd() {
    int k, i, j;
    for (k=1; k<=n; k++) {
        for (i=1; i<=n; i++) {
            for (j=1; j<=n; j++) {
                if (dist[i][j] > dist[i][k] + dist[k][j]) {
                    dist[i][j] = dist[i][k] + dist[k][j];
                }
            }
        }
    }
}

int main() {
    int u,v,w,i,j;
    scanf("%d%d", &n, &m);

    for (i=1; i<=n; i++) {
        for (j=1; j<=n; j++) {
            dist[i][j] = INF;
            g[i][j] = INF;
        }
    }


    for (i=1; i<=n; i++) {
        g[i][i] = INF;
        dist[i][i] = 0;
    }

    for (i=0; i<m; i++) {
        scanf("%d%d%d", &u, &v, &w);

        if (g[u][v] > w) {
            g[u][v] = w;
            dist[u][v] = w;
        }
    }

    floyd();

    for (i=1; i<=n; i++) {
        for (j=1; j<=n; j++) {
            printf("%ld ", dist[i][j] >= INF ? 0 : dist[i][j]);
        }
        printf("\n");
    }

    return 0;
}

```
