괄호가 두 종류이므로, 단순 카운터로는 힘들고, 실제로 스택을 구현해서 써보자

근데 입출력에 주의하고! (조건이 이상.. 여러줄에 걸쳐서 하나의 케이스 입력이 들어올수있다)
fgets는 EOF나 newline에서 끊어지는데 newline이 안들어오고 바로 EOF들어오는 케이스도 생각할것.

```
#include <stdio.h>
#include <string.h>

char stack[100] = {};
char *sp = stack;
char *pc;
char str[101];
char tmp[101];

int main() {

    while (1) {
        *str = 0;
        while (1) {
            fgets(tmp, 100, stdin);
            strncat(str, tmp, 100);
            if (tmp[strlen(tmp)-1] == '.' || tmp[strlen(tmp)-2] == '.') {
                break;
            }
        }

        if (*str == '.') {
            break;
        }

        int ok = 1;
        pc = str;
        sp = stack;
        while (*pc) {
            if (*pc == '(') {
                *sp = '(';
                sp++;
            } else if (*pc == '[') {
                *sp = '[';
                sp++;
            } else if (*pc == ')') {
                if (sp > stack && *(sp-1) == '(') {
                    sp--;
                } else {
                    ok = 0;
                    break;
                }
            } else if (*pc == ']') {
                if (sp > stack && *(sp-1) == '[') {
                    sp--;
                } else {
                    ok = 0;
                    break;
                }
            }
            pc++;
        }

        ok = ok && sp == stack;

        printf("%s\n", ok?"yes":"no");
    }

    return 0;
}

```
