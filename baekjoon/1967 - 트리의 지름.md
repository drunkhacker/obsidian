[[1167 - 트리의 지름]]과 똑같다. 입력 형식만 다름.

```c
#include <stdio.h>
#include <stdlib.h>

#if 0
#define write_log(format,args...)  printf(format, ## args)
#else
#define write_log(format,args...)
#endif

#define MAX_V 10000

typedef struct node {
    int weight;
    int vertex;
    struct node *next;
} Node;

typedef struct {
    Node *adj[MAX_V+1];
    Node *tail[MAX_V+1];
} Graph;

Graph g;

void add_edge(int u, int v, int w) {
    Node *node = malloc(sizeof(Node));
    node->next = NULL;
    node->weight = w;
    node->vertex = v;

    if (!g.adj[u]) {
        g.adj[u] = node;
        g.tail[u] = node;
    } else {
        // find tail
        Node *tail = g.tail[u];
        tail->next = node;
        g.tail[u] = node;
    }
}

int nv, mv, md;

void dfs(int u, int v, int ws) {
    // u parent
    // v current
    // w = weight(u, v)
    // ws = weight sum acc.

    if (ws > md) {
        md = ws;
        mv = v;
    }

    Node *node = g.adj[v];
    while (node) {
        if (node->vertex != u) {
            dfs(v, node->vertex, ws + node->weight);
        }
        node = node->next;
    }
}

int main() {
    int u, v, w;
    scanf("%d", &nv);
    for (int i=0; i<nv-1; i++) {
        scanf("%d %d %d", &u, &v, &w);
        add_edge(u, v, w);
        add_edge(v, u, w);
    }


    dfs(0, 1, 0);
    dfs(0, mv, 0);

    printf("%d\n", md);

    return 0;
}
```
