[[15651 - N과 M (3)]]에 monotonic increase 조건 추가니까.. 마지막으로 visit했던 친구를 넣으면 됨. [[15650 - N과 M (2)]]에 있는 visit했던 친구 넘겨주는 코드 추가.. 근데 j+1이 아니라 j여야함. 중복 허용이니!

```
#include <stdio.h>

int a[] = {0,0,0,0,0,0,0,0,0};
int v[] = {0,0,0,0,0,0,0,0,0};

void f(int n, int m, int d, int j) {
    int i;
    if (m == 0) {
        for (i=0; i<d; i++) {
            printf("%d ", a[i]);
        }
        printf("\n");
        return;
    }
    for (i=j; i<=n; i++) {
        a[d] = i;
        f(n, m-1, d+1, i);
    }
}

int main() {
    int n, m;

    scanf("%d %d", &n, &m);
    f(n, m, 0, 1);
}
```
