https://www.acmicpc.net/problem/4803

1. DFS로 연결된 노드들을 찾는다. 총 컴포넌트 갯수를 센다
2. 찾아낸 컴포넌트가 트리인지 판단을 한다.
	1. 근데 DFS로 탐색하는 와중에 판단을 할 수 있는가? - yes
	2. how? : 트리는 두 노드를 연결하는 간선이 유일해야 하므로 현재 노드에 연결된 다른 노드들 중 (부모 제외) 이미 방문한 노드가 있다면 트리의 조건에 어긋남.

```c
#include <stdio.h>
#include <string.h>

#define LOG 0
#define write_log(format,args...)        \
                  if (LOG) {                      \
                      printf(format, ## args);    \
                  }


#define N 500
int adj[N+1][N+1];
int visited[N+1];
int n, cnt;

void dfs(int p, int u, int *istree) {
    visited[u] = 1;

    // get adj
    for (int e=0; adj[u][e] != 0; e++) {
        int v = adj[u][e];

        // check is tree
        if (v != p && visited[v]) {
            *istree = 0;
        }

        if (!visited[v]) {
            dfs(u, v, istree);
        }
    }
}

int cnt_tree() {
    int cnt = 0;
    for (int v=1; v<=n; v++) {
        if (!visited[v]) {
            int istree = 1;
            write_log("starting dfs v=%d\n", v);
            dfs(0, v, &istree);
            if (istree) {
                cnt++;
            }
        }
    }

    return cnt;
}


int main() {
    int m, u, v, c=1, e;
    while (1) {
        scanf("%d %d", &n, &m);
        if (n == 0 && m == 0) {
            break;
        }

        for (int i=0; i<m; i++) {
            scanf("%d %d", &u, &v);
            for (e=0; adj[u][e] != 0; e++) { }
            adj[u][e] = v;
            for (e=0; adj[v][e] != 0; e++) { }
            adj[v][e] = u;
        }

        for (int i=1; i<=n; i++) {
            write_log("%d: ", i);
            for (int v=0; adj[i][v] != 0; v++) {
                write_log("%d ", adj[i][v]);
            }
            write_log("\n");
        }


        int cnt = cnt_tree();
        printf("Case %d: ", c++);
        if (cnt <= 1) {
            printf("%s\n", cnt == 0 ? "No trees." : "There is one tree.");
        } else {
            printf("A forest of %d trees.\n", cnt);
        }

        // reset
        memset(visited, 0, sizeof(int)*(N+1));
        memset(adj, 0, sizeof(int)*(N+1)*(N+1));
    }
    return 0;
}
```
