1차시도
```
#include <stdio.h>

int N;
int S[21][21];
int v[21] = {0,};
int min = 2147483647;
#define MIN(a, b) ((a)<(b) ? (a):(b))
#define ABS(a) ((a)>0 ? (a):(-(a)))

void backtrack(int n) {
    int i, j, s1, s2;
    if (n == N/2) {
        s1 = s2 = 0;
        for (i=0; i<N-1; i++) {
            for (j=i+1; j<N; j++) {
                if (v[i] && v[j]) {
                    s1 += (S[i][j] + S[j][i]);
                }

                if (!v[i] && !v[j]) {
                    s2 += (S[i][j] + S[j][i]);
                }
            }
        }

        min = MIN(min, ABS(s1-s2));
        return;
    }

    for (i=n; i<N; i++) {
        if (!v[i]) {
            v[i] = 1;
            backtrack(n+1);
            v[i] = 0;
        }
    }
}

int main() {
    int i, j;
    scanf("%d", &N);

    for (i=0; i<N; i++) {
        for (j=0; j<N; j++) {
            scanf("%d", &S[i][j]);
        }
    }

    backtrack(0);

    printf("%d\n", min);
}
```

n번째 문제는, n번째 인원을 1팀에 넣느냐 0팀에 넣느냐 니까, 
f(n, c) =  // c는 1팀의 인원수
  1팀에 넣으면, v\[n\] = 1, f(n+1, c+1)
  0팀에 넣으면, v\[n\] = 0, f(n+1, c)

pruning 조건: 만약 각 팀의 인원이 이미 과반을 넘어버리면 invalid한 state이므로 버린다

```
#include <stdio.h>

int N;
int S[21][21];
int v[21] = {0,};
int min = 2147483647;
#define MIN(a, b) ((a)<(b) ? (a):(b))
#define ABS(a) ((a)>0 ? (a):(-(a)))

void backtrack(int n, int c) {
    int i, j, s1, s2;
    if (c == N/2) {
        s1 = s2 = 0;
        for (i=0; i<N-1; i++) {
            for (j=i+1; j<N; j++) {
                if (v[i] && v[j]) {
                    s1 += (S[i][j] + S[j][i]);
                }

                if (!v[i] && !v[j]) {
                    s2 += (S[i][j] + S[j][i]);
                }
            }
        }
        for (i=0; i<N; i++) {
            if (v[i]) {
                printf("1 ");
            } else {
                printf("0 ");
            }
        }
        printf("\n");
        min = MIN(min, ABS(s1-s2));
        return;
    }


    v[n] = 1;
    if (c < N/2) {
        backtrack(n+1, c+1);
    }


    v[n] = 0;
    if (n - c < N/2) {
        backtrack(n+1, c);
    }
}

int main() {
    int i, j;
    scanf("%d", &N);

    for (i=0; i<N; i++) {
        for (j=0; j<N; j++) {
            scanf("%d", &S[i][j]);
        }
    }

    backtrack(0, 0);

    printf("%d\n", min);
    return 0;
}


```

tip: 중간에 팀 배열이 제대로 됐는지 디버그 프린트 하는게 유용한듯.
