BFS로 최단거리 구하기
```
#include <stdio.h>

#define N 300

typedef struct point {
    int x;
    int y;
} Point;

int visited[N+1][N+1];
Point q[N*N];
int qhead, qtail;

void check_enqueue(int x, int y, Point p, int n) {
    if (y >= 0 && y < n && x >= 0 && x < n && visited[x][y] == 0) {
        visited[x][y] = visited[p.x][p.y]+1;
        q[qtail++] = (Point){x, y};
    }
}

int main() {
    int t, n;
    scanf("%d", &t);
    while (t--) {
        int sx, sy, tx, ty;
        scanf("%d", &n);
        scanf("%d%d", &sx, &sy);
        scanf("%d%d", &tx, &ty);

        visited[sx][sy] = 1;
        q[qtail++] = (Point){sx, sy};
        while (qhead != qtail && visited[tx][ty] == 0) {
            Point u = q[qhead++];

            check_enqueue(u.x-2, u.y-1, u, n);
            check_enqueue(u.x-1, u.y-2, u, n);
            check_enqueue(u.x+1, u.y-2, u, n);
            check_enqueue(u.x+2, u.y-1, u, n);
            check_enqueue(u.x+2, u.y+1, u, n);
            check_enqueue(u.x+1, u.y+2, u, n);
            check_enqueue(u.x-1, u.y+2, u, n);
            check_enqueue(u.x-2, u.y+1, u, n);
        }

        printf("%d\n", visited[tx][ty] - 1);

        // teardown
        qtail = qhead = 0;
        for (sx=0; sx<n; sx++) 
            for (sy=0; sy<n; sy++)
                visited[sx][sy] = 0;
    }

    return 0;
}
```
