https://www.acmicpc.net/problem/7576

중간에 이미 visit한 노드여도 predecessor 값 + 1 한게 기존값보다도 적으면 다시 살펴볼 필요가 있다.

```
#include <stdio.h>

#define N 1000

int map[N][N];
int visited[N][N];
int m, n;
int qhead, qtail;
int c;

typedef struct {
    int x;
    int y;
} Point;

#define LOG 0
#define write_log(format,args...)        \
                  if (LOG) {                      \
                      printf(format, ## args);    \
                  }


#define N_Q (N*N)
Point q[N_Q];

#define visit_mark(u,x,y) do { \
    if (y >= 0 && y < n && x >= 0 && x < m && map[y][x] == 0) {\
        if (visited[y][x] == 0) {\
            visited[y][x] = visited[u.y][u.x] + 1;\
            c = visited[y][x];\
            q[qtail] = (Point){x,y};\
            qtail = (qtail+1)%N_Q;\
            write_log("visit (%d,%d)=%d\n",x,y,c);\
        } else {\
            if (visited[u.y][u.x] + 1 < visited[y][x])  {\
                visited[y][x] = visited[u.y][u.x] + 1;\
                c = visited[y][x];\
                q[qtail] = (Point){x,y};\
                qtail = (qtail+1)%N_Q;\
                write_log("update (%d,%d)=%d\n",x,y,c);\
            }\
        }\
    }\
} while(0);

void bfs() {
    int i,j;
    // initial positions
    for (i=0; i<n; i++) {
        for (j=0; j<m; j++) {
            if (map[i][j] == 1) {
                visited[i][j] = 1;
                write_log("visit (%d,%d)=%d\n", j,i,1);
                q[qtail] = (Point){j, i};
                qtail = (qtail+1)%N_Q;
            }
        }
    }

    while (qhead != qtail) {
        Point u = q[qhead];
        write_log("qhead=%d, (%d,%d)\n", qhead, u.x, u.y);
        qhead = (qhead+1)%N_Q;

        int x,y;
        y = u.y-1; x = u.x;
        visit_mark(u,x,y);

        y = u.y+1; x = u.x;
        visit_mark(u,x,y);

        y = u.y; x = u.x-1;
        visit_mark(u,x,y);

        y = u.y; x = u.x+1;
        visit_mark(u,x,y);
    }


}

int main() {
    int i, j;
    scanf("%d%d", &m, &n);
    for (i=0; i<n; i++) {
        for (j=0; j<m; j++) {
            scanf("%d", &map[i][j]);
        }
    }

    bfs();

    // check all covered
    c = 1;
    for (i=0; i<n; i++) {
        for (j=0; j<m; j++) {
            if (visited[i][j] == 0 && map[i][j] != -1) {
                printf("-1\n");
                return 0;
            } else if (visited[i][j] > c) {
                c = visited[i][j];
            }
        }
    }
    printf("%d\n", c-1);

    return 0;
}
```
