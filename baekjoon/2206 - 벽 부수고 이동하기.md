BFS 최단경로 문제이긴 한데, 중간에 인접 노드를 탐색하다가 벽을 발견하면 그걸 없애는 경우를 넣으면서 다녀야할듯.. 한데,
벽을 없애고 난 후에 이게 아니다 싶었을때 복구는 어떻게 하지?
인접한 벽을 없애기로 한 칸을 스택에 넣고 나중에 거기서부터 다시 BFS.
bfs 하는 도중에 사방을 봤을때 벽에 막힌 지점들을 스택에 넣고 BFS를 한턴 끝낸 다음에, 

스택에 있는 애들을 하나씩 꺼내면서 벽을 없애는 시도...?

----
열심히 찾다보니 3차원배열을 쓰라는 힌트가 있음.

`visit[x][y][0]`을 벽을 하나도 부수지 않고 이동하면서 x,y로 도착했을때 최단경로라고 하자.
`visit[x][y][1]`은 x,y가 벽일때 부쉈을때 최단경로. 부쉈으니까 이제 더 이상 부술수가 없고..
그럼 그 담은? (x,y)가 벽이었음에도 불구하고 q에 넣으면 그 담부터는 벽이 아닌 인접 노드만 가져가면 됨.

pop u -> (x,y)에서 다음 인접노드들을 집어넣을때 x,y가
- 벽이 아니라면 visit\[x\]\[y\]\[u.z\] = visit\[u.x\]\[u.y\]\[u.z\] + 1
- 벽을 만났고 
	- 그 벽을 아직 안부쉈다면 visit\[x\]\[y\]\[1\] = visit\[u.x\]\[u.y\]\[0\] + 1
	- 아무것도 못함.

그 후 q에서 노드 꺼냈을때 (x,y,z) z = 0이라면 아직 벽을 안부셨다는 뜻이고,
z =1 이라면 벽을 이미 부순 상태다. 

참고 https://dogsavestheworld.tistory.com/249 여기서 소스코드 보고 힌트를 얻음 ㅜㅜ 

z = 1을 선택한 이후에는 무조건 z=1라인을 타게 한다. 
z = 0인 경우에는 인접 노드가 벽인 케이스면 z=1로 세팅하고 벽의 포인트를 enqueue한다.

그리고 남 코드를 보면서 깨달은 그간 안좋은 습관 중 하나는, 
1. bfs자체가 타겟 노드에 항상 최단 거리로 접근하기때문에 거리 update케이스를 한번 더 고민할 필요가 없다. 즉 x,y visit 했다고 마킹됐을때 내 기존 코드에서는 visit(x)(y)가 좀 더 나아질 경우를 고민했다면, 이미 bfs의 특성상 최단거리로 들어오기 때문에 update 과정이 필요가 없다. 
   아마 update를 시도했던건 [[7569 - 토마토]]에서처럼 BFS시작점이 여러개여서 특정 포인트에 도달하는 경우가 많을때와 [[16928 - 뱀과 사다리 게임]]에서처럼 특수한 연결 케이스가 있었을때 고려했어야하는걸 그냥 복사 붙여넣기 했어서 그런것 같다.
2. dequeue하다가 타겟 노드에 다다르면 바로 튀어나가면 된다. 굳이 q가 빌때까지 계속 돌릴 필요는 없다. 

```c
#include <stdio.h>
#define MAX_N 1000
#define MAX_M 1000

#define LOG 0
#define write_log(format,args...)        \
                  if (LOG) {                      \
                      printf(format, ## args);    \
                  }


int N, M;
int map[MAX_N][MAX_M];
typedef struct {
    int x;
    int y;
    int z;
} Point;


Point q[10*MAX_N*MAX_M];
int qhead;
int qtail;
int wall_broken = 0;

int visit[MAX_N][MAX_M][2];

void check_mark(Point u, int x, int y) {
    if (!(x >= 0 && x < M && y >= 0 && y < N))  {
        return ;
    }

    if (map[y][x] == 0) {
        if (visit[y][x][u.z] == 0) {
            visit[y][x][u.z] = visit[u.y][u.x][u.z] + 1;
            q[qtail++] = (Point){.x = x, .y = y, .z = u.z};
            write_log("cm visit (%d,%d,%d) = %d\n", y, x, u.z, visit[y][x][u.z]);
        }
    } else if (u.z == 0) {
        visit[y][x][1] = visit[u.y][u.x][0] + 1;
        q[qtail++] = (Point){.x = x, .y = y, .z = 1};
        write_log("destroy wall & visit (%d,%d,%d) = %d\n", y, x, 1, visit[y][x][1]);
    }
}

int bfs(Point s, int v) {
    write_log("bfs visit (%d,%d,%d) = %d\n", s.y, s.x, s.z, v);
    visit[s.y][s.x][s.z] = v;
    q[qtail++] = s;

    while (qhead != qtail) {
        Point u = q[qhead++];
        if (u.x == M-1 && u.y == N-1) {
            return visit[u.y][u.x][u.z];
        }
        int x = u.x + 1; int y = u.y;
        check_mark(u, x, y);
        x = u.x - 1; y = u.y;
        check_mark(u, x, y);
        x = u.x; y = u.y + 1;
        check_mark(u, x, y);
        x = u.x; y = u.y - 1;
        check_mark(u, x, y);
    }
    return -1;
}

int main() {
    int i, j;
    char c;

    scanf("%d%d", &N, &M);
    while ((getchar()) != '\n');
    
    for (i=0; i<N; i++) {
        for (j=0; j<M; j++) {
            map[i][j] = getchar() - '0';
        }
        while ((c = getchar()) != '\n' && c != EOF);
    }

    // init pass
    int s = bfs((Point){.x = 0, .y = 0, .z = 0}, 1);

    printf("%d\n", s);

    return 0;
}
```
