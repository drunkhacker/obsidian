dijkstra 구현.
priority queue를 구현할 때 heap으로 구현 시 percolate down구현을 잘못하면 NPE가 난다..
그리고 graph 표현 시, 인접 노드 리스트가 길어지면 끄트머리를 탐색하다가 timeout이 나므로 tail을 따로 저장해두는게 좋을듯.

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define MAX_V 20000
#define MAX_E 300000
#define INF 1e9

int dist[MAX_V+1];

#define LOG 0
#define write_log(format,args...)        \
                  if (LOG) {                      \
                      printf(format, ## args);    \
                  }

typedef struct node {
    int weight;
    int vertex;
    struct node *next;
} Node;

typedef struct {
    int size;
    Node *adj[MAX_V+1];
    Node *tail[MAX_V+1];
} Graph;

Graph g;

void add_edge(int u, int v, int w) {
    Node *node = malloc(sizeof(Node));
    node->next = NULL;
    node->weight = w;
    node->vertex = v;

    if (!g.adj[u]) {
        g.adj[u] = node;
        g.tail[u] = node;
    } else {
        // find tail
        Node *tail = g.tail[u];
        tail->next = node;
        g.tail[u] = node;
    }
}

// priority queue

typedef struct {
    int prio;
    int elem;
} QElem;
QElem *q[MAX_V*2];
int q_size;

void percolateDown(int i) {
    int li = 2*i+1, ri = 2*i+2;

    if (li >= q_size) {
        return;
    }

    int target;
    if (ri >= q_size || q[li]->prio < q[ri]->prio) {
        target = li;
    } else {
        target = ri;
    }

    if (q[target]->prio < q[i]->prio) {
        QElem *t = q[i];
        q[i] = q[target];
        q[target] = t;
        percolateDown(target);
    }
}

void percolateUp(int i) {
    if (i == 0) {
        return;
    }
    int pi = (i-1)/2;
    // check is heap
    if (q[pi]->prio > q[i]->prio) {
        QElem *t = q[i];
        q[i] = q[pi];
        q[pi] = t;
        percolateUp(pi);
    }
}

QElem* pop() {
    if (q_size <= 0) {
        return NULL;
    }
    QElem *t = q[0];
    q[0] = q[q_size-1];
    q[q_size-1] = NULL;
    q_size--;
    percolateDown(0);
    return t;
}

void insert(QElem *x) {
    // insert to bin tree
    q[q_size] = x;
    percolateUp(q_size++);
}

void dijkstra(int src) {
    for (int v=1; v<=g.size; v++) {
        dist[v] = INF;
    }

    dist[src] = 0;
    QElem *e = malloc(sizeof(QElem));
    e->elem = src;
    e->prio = 0;
    insert(e);

    while (q_size > 0) {
        QElem *u = pop();

        if (dist[u->elem] < u->prio) {
            continue;
        }

        Node *n = g.adj[u->elem];
        write_log("adj of vertex %d\n", u->elem);
        while (n) { // for all adj nodes of u
            int v = n->vertex;
            int w = n->weight;
            write_log("%d->%d , w=%d\n", u->elem, v, dist[v]);
            if (dist[v] > dist[u->elem] + w) {
                dist[v] = dist[u->elem] + w;


                write_log("update dist to %d to %d\n", v, dist[v]);

                e = malloc(sizeof(QElem));
                e->elem = v;
                e->prio = dist[v];
                insert(e);
            }

            n = n->next;
        }
    }
}

int main() {
    int nv,ne,i,src,u,v,w;

    scanf("%d%d%d", &nv, &ne, &src);
    g.size = nv;
    for (i=0; i<ne; i++) {
        scanf("%d%d%d", &u,&v,&w);
        add_edge(u,v,w);
    }

    dijkstra(src);

    for (i=1; i<=nv; i++) {
        printf(dist[i] == INF ? "INF\n" : "%d\n", dist[i]);
    }

    return 0;
}
```
