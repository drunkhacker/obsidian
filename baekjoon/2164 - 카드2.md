간단한 circular q 구현

```c
#include <stdio.h>

#define N (500000 + 10)
int q[N];
int rear = -1, front = -1;

int isfull() {
    return (front == rear + 1) || (front == 0 && rear == N-1);
}

int isempty() {
    return front == -1;
}

int size() {
    if (isfull()) { 
        return N;
    } else if (isempty()) {
        return 0;
    } else {
        return rear >= front ? rear - front + 1 : rear - front + N + 1;
    }
}

void enq(int elem) {
    if (isfull()) {
        // do nothing
    } else {
        if (front == -1) {
            front = 0;
        }
        rear = (rear + 1)%N;
        q[rear] = elem;
    }
}

int deq() {
    if (isempty()) {
        return -1;
    } else {
        int elem;
        elem = q[front];
        if (front == rear) {
            front = rear = -1;
        } else {
            front = (front+1)%N;
        }

        return elem;
    }
}

int main() {
    int n, i;
    scanf("%d", &n);
    for (i=1; i<=n; i++) {
        enq(i);
    }

    while (size() > 1) {
        deq();
        if (size() == 1) {
            break;
        }
        enq(deq());
    }

    printf("%d\n", q[front]);
    return 0;
}
```
