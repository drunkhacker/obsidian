$$
\begin{bmatrix}
F_n \\
F_{n-1}
\end{bmatrix}
 = 
 \begin{bmatrix}
1 & 1 \\
1 & 0
\end{bmatrix}
\begin{bmatrix}
F_{n-1} \\
F_{n-2}
\end{bmatrix}
$$
이므로, $\begin{pmatrix}1&1\\1&0\end{pmatrix}$을 n제곱 하면 n번째 fibonacci를 얻을수 있게 된다. (물론 인덱싱이 좀 틀릴순 있음)

```
#include <stdio.h>
#define MOD 1000000007

int a[2][2] = {{1, 1}, {1, 0}};
int b[2][2];

void mul(int (*a)[2], int (*b)[2], int (*c)[2], int sz) { // c = a * b
    int i,j,k;
    for (i=0; i<sz; i++) {
        for (j=0; j<sz; j++) {
            c[i][j] = 0;

            for (k=0; k<sz; k++) {
                c[i][j] += ((long)a[i][k]*b[k][j]) % MOD;
            }

            c[i][j] %= MOD;
        }
    }
}

void mpow(int (*a)[2], int (*b)[2], long p, int n) { // b = a^p
    int i,j;
    int t[2][2];
    if (p == 1) {
        for (i=0; i<n; i++) {
            for (j=0; j<n; j++) {
                b[i][j] = a[i][j] % MOD;
            }
        }
        return;
    }

    if (p % 2) {
        mpow(a, t, p-1, n);
        mul(a, t, b, n);
    } else {
        mpow(a, t, p/2, n);
        mul(t, t, b, n);
    }
}

int main() {
    long n;
    scanf("%ld", &n);
    if (n < 3) {
        printf("%d\n", n == 0 ? 0 : n == 1 ? 1 : n == 2 ? 1 : 2);

    } else {
        mpow(a, b, n - 2, 2);
        printf("%d\n", (b[0][0] + b[0][1]) % MOD);
    }

    return 0;
}
```
