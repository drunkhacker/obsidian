q에 일단 1~N까지 넣은다음에 K-1번 pop을 하면서 나오는 애는 뒤로 보내고 K번째 pop을 할때 나오는 숫자가 요세푸스 순열의 next 원소가 된다.

ex) (7,3)-요세푸스

1 2 3 4 5 6 7
-- pop하고 1 뒤로
2 3 4 5 6 7 1
-- pop하고 2 뒤로
(3) 4 5 6 7 1 2
-- pop하고 3 내놓음
5 6 7 1 2 4 
-- pop하고 4뒤로
6 7 1 2 4 5
-- pop하고 5뒤로 
(6) 7 1 2 4 5 
-- pop하고 6 내놓음
1 2 4 5 7
-- pop하고 7뒤로
2 4 5 7 1
-- pop하고 1뒤로
(2) 4 5 7 1
-- pop하고 2내놓음
4 5 7 1
-- pop하고 4뒤로 
5 7 1 4
-- pop하고 5뒤로
(7) 1 4 5
-- pop하고 7내놓음
1 4 5
-- pop하고 1뒤로
4 5 1
-- pop하고 4뒤로
(5) 1 4
-- pop하고 5내놓음
1 4 
-- pop하고 1뒤로
4 1 
-- pop하고 4뒤로
(1) 4 
...

```
#include <stdio.h>
#define N 1001

struct queue {
    int arr[N];
    int st, en;
} q;

void push(struct queue *q, int v) {
    q->arr[q->en] = v;
    q->en = (q->en + 1) % N;
}

int pop(struct queue *q) {
    int r = q->arr[q->st];
    q->st = (q->st + 1) % N;
    return r;
}

int empty(struct queue *q) {
    return q->st == q->en;
}

int main() {
    int n, k, i, x;
    scanf("%d %d", &n, &k);

    for (i=1; i<=n; i++) {
        push(&q, i);
    }

    i = 0;
    printf("<");
    while (!empty(&q)) {
        x = pop(&q);
        i++;
        if (i % k == 0) {
            if (i > k) {
                printf(", ");
            }
            printf("%d", x);
        } else {
            push(&q, x);
        }
    }
    printf(">\n");

    return 0;
}
```
