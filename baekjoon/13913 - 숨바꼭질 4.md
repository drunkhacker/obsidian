[[13549 - 숨바꼭질 3]]과 비슷한데 u에서 뻗어나갈 수 있는 다음 스텝 x가 있을때, x의 predecessor가 u임을 `b[]`에 저장.  `b[u] = x` 로 저장하면 u에서 뻗어나갈 수 있는 스텝이 많기 때문에 안되고, `b[x] = u` 형태로 저장해야함.

그렇기 때문에 나중에 print할때 stack을 이용해서 (여기서는 재귀) 프린트하여야 함.

```c
#include <stdio.h>

#define LOG 0
#define write_log(format,args...)        \
                  if (LOG) {                      \
                      printf(format, ## args);    \
                  }

#define N 100001
int visit[N];
int b[N];

#define NQ (3*N)
int q[NQ];
int qtail, qhead;

void check_mark(int u, int x, int d) {
    if (x >= 0 && x < N && (visit[x] == 0 || visit[x] > visit[u] + d)) {
        visit[x] = visit[u] + d;

        write_log("%d -> %d (d=%d)\n", u, x, d);

        // u -> x
        b[x] = u;

        q[qtail] = x;
        qtail = (qtail +1)%NQ;
    }
}

void print_back(int k, int n) {
    if (k != n)
        print_back(b[k], n);
    printf("%d ", k);
}

int main() {
    int n, k;
    scanf("%d%d", &n, &k);

    visit[n] = 1;
    q[qtail] = n;
    qtail = (qtail+1)%NQ;

    while (qhead != qtail) {
        int u = q[qhead];
        qhead = (qhead+1)%NQ;

        int x;
        x = u-1;
        check_mark(u, x, 1);

        x = u+1;
        check_mark(u, x, 1);

        x = 2*u;
        check_mark(u, x, 1);

    }

    printf("%d\n", visit[k] - 1);
    print_back(k, n);
    printf("\n");

    return 0;
}

```
