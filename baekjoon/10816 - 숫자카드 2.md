https://www.acmicpc.net/problem/10816
first occurence와 last occurence를 이분탐색으로 찾아서 인덱스 차이로 구하기
```
#include <stdio.h>
#include <stdlib.h>

int a[500000];

int comp(const void *a, const void *b) {
    return (*(int *)a) > (*(int *)b) ? 1 : (*(int *)a) < (*(int *)b) ? -1 : 0;
}

int solve(int n, int x) {
    int i, j, m, st, en;
    // see first occur
    i = -1;
    st = 0; en = n-1;
    while (st <= en && i == -1) {
        m = (st + en)/2;
        if (a[m] == x && (m == 0 || a[m-1] < x)) {
            i = m;
            break;
        } else if (x > a[m]) {
            st = m + 1;
        } else {
            en = m - 1;
        }
    }

    if (i < 0) {
        return 0;
    }

    // see last occur
    j = -1;
    st = i; en = n-1;
    while (st <= en && j == -1) {
        m = (st + en)/2;
        if (a[m] == x && (m == n-1 || a[m+1] > x)) {
            j = m;
            break;
        } else if (x < a[m]) {
            en = m - 1;
        } else {
            st = m + 1;
        }
    }

    return j - i + 1;
}

int main() {
    int n, m, i, x;

    scanf("%d", &n);
    for (i=0; i<n; i++) {
        scanf("%d", &a[i]);
    }

    qsort(a, n, sizeof(int), comp);

    scanf("%d", &m);
    while (m--) {
        scanf("%d", &x);
        printf("%d ", solve(n, x));
    }
    printf("\n");

    return 0;
}
```
