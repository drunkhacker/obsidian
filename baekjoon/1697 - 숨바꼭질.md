BFS 최단거리 문제인데 1차원에서. 3\*N 이라는 q 사이즈는 경우의 수가 3개라 그런건데 딱히 valid하지는 않음.



```
#include <stdio.h>

#define N 100001

int visit[N];
#define NQ (3*N)
int q[NQ];
int qtail, qhead;

void check_mark(int u, int x) {
    if (x >= 0 && x < N && (visit[x] == 0 || visit[x] > visit[u]+1)) {
        visit[x] = visit[u] + 1;
        q[qtail] = x;
        qtail = (qtail+1)%NQ;
    }
}

int main() {
    int n,k;
    scanf("%d%d", &n, &k);

    visit[n] = 1;

    q[qtail] = n;
    qtail = (qtail+1)%NQ;

    while (qhead != qtail) {
        int u = q[qhead];
        qhead = (qhead+1)%NQ;

        int x;
        x = u-1;
        check_mark(u,x);

        x = u+1;
        check_mark(u,x);

        x = 2*u;
        check_mark(u,x);
    }

    printf("%d\n", visit[k] - 1);

    return 0;
}

```
