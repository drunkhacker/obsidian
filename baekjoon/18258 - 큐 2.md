https://www.acmicpc.net/problem/18258

ring q에서 size 함수는 어떻게 구현하는가..
en > st -> en - st
en < st -> N - (st - en)
en == st -> 0

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define N 2000000

struct queue {
    int arr[N];
    int st, en;
} q;

void push(struct queue *q, int v) {
    q->arr[q->en] = v;
    q->en = (q->en + 1) % N;
}

int size(struct queue *q) {
    int s = q->en - q->st;
    return s < 0 ? N + s : s;
}

int pop(struct queue *q) {
    int r;
    if (size(q) == 0) {
        return -1;
    }

    r = q->arr[q->st];
    q->st = (q->st + 1) % N;
    return r;
}

int front(struct queue *q) {
    if (size(q) == 0) {
        return -1;
    }

    return q->arr[q->st];
}

int back(struct queue *q) {
    if (size(q) == 0) {
        return -1;
    }

    if (q->en == 0) {
        return q->arr[N-1];
    }

    return q->arr[q->en-1];
}

int main() {
    int n, i, x;
    char *pc;
    char cmd[20];

    // init q
    q.en = 0;
    q.st = 0;

    scanf("%d", l&n);
    while ((getchar()) != '\n');
    for (i=0; i<n; i++) {
        fgets(cmd, 19, stdin);
        if (strncmp(cmd, "push", 4) == 0) {
            pc = cmd;
            strsep(&pc, " ");
            x = atoi(strsep(&pc, " "));
            push(&q, x);
        } else if (strncmp(cmd, "pop", 3) == 0) {
            printf("%d\n", pop(&q));
        } else if (strncmp(cmd, "size", 4) == 0) {
            printf("%d\n", size(&q));
        } else if (strncmp(cmd, "empty", 5) == 0) {
            printf("%d\n", size(&q) == 0 ? 1 : 0);
        } else if (strncmp(cmd, "front", 5) == 0) {
            printf("%d\n", front(&q));
        } else if (strncmp(cmd, "back", 4) == 0) {
            printf("%d\n", back(&q));
        } else {
            printf("cannot parse cmd=%s\n", cmd);
        }
    }

    return 0;
}
```
