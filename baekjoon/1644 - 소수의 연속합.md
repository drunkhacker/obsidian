소수를 판별하는 조금 효율적인 방법을 이용하였음.
https://en.wikipedia.org/wiki/Primality_test

그 후는 [[1912 - 연속합]]과 같은 투포인터.

```c
#include <stdio.h>

#define LOG 0
#define write_log(format,args...)        \
                  if (LOG) {                      \
                      printf(format, ## args);    \
                  }

int p[4000000];

int isPrime(int n)
{
    if (n == 2 || n == 3)
        return 1;

    if (n <= 1 || n % 2 == 0 || n % 3 == 0)
        return 0;

    for (int i = 5; i * i <= n; i += 6) {
        if (n % i == 0 || n % (i + 2) == 0)
            return 0;
    }

    return 1;
}


int setprimes(int n) {
    int i;
    int np = 0;
    for (i=2; i<=n; i++) {
        if (isPrime(i)) {
            p[np++] = i;
        }
    }

    return np;
}

int main() {
    int n, np;
    scanf("%d", &n);

    np = setprimes(n);

    int s = 0, e = 0, cnt = 0;
    int psum = p[0];

    while (s <= e && e < np) {
        if (psum < n) {
            e++;
            psum += p[e];
        } else { 
            if (psum == n) {
                write_log("s=%d, e=%d\n", p[s], p[e]);
                cnt++;
            }
            psum -= p[s];
            s++;
        }
    }

    printf("%d\n", cnt);

    return 0;
}
```
