#include <stdio.h>
#include <string.h>
#include <limits.h>


#define LOG 0
#define write_log(format,args...)        \
                  if (LOG) {                      \
                      printf(format, ## args);    \
                  }


struct heap {
    int size;
    int a[1<<17];
    int (*isheap)(int, int);
} left, right;

typedef struct heap Heap;

int is_minheap(int p, int c) {
    return p < c;
}

int is_maxheap(int p, int c) {
    return p > c;
}

void percolateDown(Heap *heap, int i) {
    int li = 2*i+1, ri = 2*i+2, t;

    if (i >= heap->size || (heap->a[li] == INT_MAX && heap->a[ri] == INT_MAX)) {
        return;
    }

    if ((heap->a[li] != INT_MAX && !heap->isheap(heap->a[i], heap->a[li])) || 
            (heap->a[ri] != INT_MAX && !heap->isheap(heap->a[i], heap->a[ri]))) {

        if (heap->a[li] != INT_MAX && heap->a[ri] != INT_MAX) {
            if (heap->isheap(heap->a[li], heap->a[ri])) {
                t = heap->a[li];
                heap->a[li] = heap->a[i];
                heap->a[i] = t;
                percolateDown(heap, li);
            } else {
                t = heap->a[ri];
                heap->a[ri] = heap->a[i];
                heap->a[i] = t;
                percolateDown(heap, ri);
            }
        } else if (heap->a[li] != INT_MAX) {
            t = heap->a[li];
            heap->a[li] = heap->a[i];
            heap->a[i] = t;
            percolateDown(heap, li);
        } else if (heap->a[ri] != INT_MAX) {
            t = heap->a[ri];
            heap->a[ri] = heap->a[i];
            heap->a[i] = t;
            percolateDown(heap, ri);
        }
    }
}

void percolateUp(Heap* heap, int i) {
    if (i == 0) {
        return;
    }

    int pi = (i-1)/2;
    if (!heap->isheap(heap->a[pi], heap->a[i])) {
        int t = heap->a[i];
        heap->a[i] = heap->a[pi];
        heap->a[pi] = t;
        percolateUp(heap, pi);
    }
}


void init(Heap *heap, int (*isheap)(int, int)) {
    heap->size = 0;
    heap->isheap = isheap;
    for (int i=0; i<sizeof(heap->a) / sizeof(int); i++) {
        heap->a[i] = INT_MAX;
    }
}

int peak(Heap *heap) {
    return heap->a[0];
}

int pop(Heap *heap) {
    if (heap->size <= 0) {
        return 0;
    }

    int t = heap->a[0];
    heap->a[0] = heap->a[heap->size-1];
    heap->a[heap->size-1] = INT_MAX;
    heap->size--;
    percolateDown(heap, 0);
    return t;
}

void insert(Heap *heap, int x) {
    heap->a[heap->size] = x;
    percolateUp(heap, heap->size);
    heap->size++;
}

void print_heaps() {
    int i;
    write_log("left = ");
    for (i=0; i<left.size; i++) {
        write_log("%d, ", left.a[i]);
    }
    write_log("\nright = ");
    for (i=0; i<right.size; i++) {
        write_log("%d, ", right.a[i]);
    }
    write_log("\n");
}

int main() {
    int n, x;

    init(&left, is_maxheap);
    init(&right, is_minheap);

    scanf("%d", &n);
    while (n--) {
        scanf("%d", &x);
        // insert to median heap
        if (right.size > 0 && peak(&right) < x) {
            write_log("insert %d to right\n", x);
            insert(&right, x);
            if (right.size - left.size > 1) {
                write_log("pop %d from right and insert to left\n", peak(&right));
                insert(&left, pop(&right));
            }
        } else {
            write_log("insert %d to left\n", x);
            insert(&left, x);
            if (left.size - right.size > 1) {
                write_log("pop %d from left and insert to right\n", peak(&left));
                insert(&right, pop(&left));
            }
        }
        print_heaps();

        // peak
        if (left.size >= right.size) {
            printf("%d\n", peak(&left));
        } else {
            printf("%d\n", peak(&right));
        }
    }

    return 0;
}
