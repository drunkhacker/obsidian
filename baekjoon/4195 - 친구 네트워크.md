스트링 해시 만드는게 불편해서 kotlin으로 풀었다. [[1717 - 집합의 표현]]과 같은 유니온 파인드 문제인데, 유니온 할때 집합의 크기를 업데이트 할것.

```kotlin
import java.util.Scanner

lateinit var parent: Array<Int>
lateinit var size: Array<Int>
lateinit var names: MutableMap<String, Int>
var ncnt = 0

fun find(a: Int): Int {
    if (parent[a] == a) {
        return a
    }

    val r = find(parent[a])
    parent[a] = r
    return r
}

fun union(a: Int, b: Int): Int {
    val ra = find(a)
    val rb = find(b)

    if (ra != rb) {
        parent[rb] = ra
        size[ra] += size[rb]
    }
    return size[ra]
}

fun main() {
    val sc = Scanner(System.`in`)

    val t = sc.nextInt()
    (1..t).forEach {
        val f = sc.nextInt()
        parent = Array(200001) { it }
        size = Array(200001) { 1 }
        names = mutableMapOf()
        ncnt = 0
        (1..f).forEach {
            val namea = sc.next()
            val nameb = sc.next()

            names[namea] = names[namea] ?: ncnt++
            val a = names[namea]!!

            names[nameb] = names[nameb] ?: ncnt++
            val b = names[nameb]!!

            println(union(a, b))
        }
    }
}
```
