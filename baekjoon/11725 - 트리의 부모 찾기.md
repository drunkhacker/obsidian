1번 노드부터 dfs를 수행해가면서 `visit[]`어레이를 parent를 저장하는 용도로 쓰자.

```c
#include <stdio.h>

#define LOG 0
#define write_log(format,args...)        \
                  if (LOG) {                      \
                      printf(format, ## args);    \
                  }



#define N 100000

typedef struct node Node;
struct node {
    int n;
    Node* next;
};

typedef struct graph Graph;
struct graph {
    int size; // # of nodes
    Node *adj[N+1];
    Node *tails[N+1];
} g;


Node node_pool[2*N];
int pc;

int visited[N+1];

Node *get_new_node() {
    return &node_pool[pc++];
}

void add_edge(int u, int v) {
    write_log("add edge %d->%d\n", u, v);
    Node *node = get_new_node();
    node->next = NULL;
    node->n = v;

    if (!g.adj[u]) {
        g.adj[u] = node;
    } else {
        g.tails[u]->next = node;
    }
    g.tails[u] = node;
}

void dfs(int u, int v) {
    write_log("visit %d, parent %d\n", v, u);
    visited[v] = u;

    Node *node = g.adj[v];
    while (node) {
        if (!visited[node->n]) {
            dfs(v, node->n);
        }
        node = node->next;
    }
}

int main() {
    int n, u, v, i;

    scanf("%d", &n);

    for (i=0; i<n-1; i++) {
        scanf("%d %d", &u, &v);
        add_edge(u, v);
        add_edge(v, u);
    }

    dfs(1, 1);

    for (v=2; v<=n; v++) {
        printf("%d\n", visited[v]);
    }

    return 0;
}

```
