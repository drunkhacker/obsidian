https://www.acmicpc.net/problem/1932

dp\[i\]\[j\] = (i, j)를 올때까지 경로 합의 최댓값..
dp\[i\]\[j\] = max(dp\[i-1\]\[j-1\], dp\[i-1\]\[j\]) + a\[i\]\[j\]

```
#include <stdio.h>
#define max(a,b) ((a)>(b)?(a):(b))

int a[500][500] = {0, };
int dp[500][500] = {0, };

int main() {
    int n, i, j, m;

    scanf("%d", &n);
    for (i=0; i<n; i++) {
        for (j=0; j<=i; j++) {
            scanf("%d", &a[i][j]);
        }
    }

    for (i=0; i<n; i++) {
        for (j=0; j<=i; j++) {
            dp[i][j] = max(dp[i-1][j-1], dp[i-1][j]) + a[i][j];
        }
    }

    m = 0;
    for (i=0; i<n; i++) {
        m = max(m, dp[n-1][i]);
    }

    printf("%d\n", m);
    return 0;
}

```
