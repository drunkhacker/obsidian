쿼드트리 

구현을 array로 하는건 binary 트리보다 훨씬 인덱싱이 복잡하기도 하고 공간적인 낭비가 심할 수 있어서 추천을 안하는듯..

bound 네개를 정하고, (top left와 bottom right로 규정)
각 네 자식 노드를 UL, UR, BL, BR.
value는 0 혹은 1

트리 탐색이야 dfs로 하면 되겠지..

```
#include <stdio.h>
#include <stdlib.h>

int a[128][128] = {0,};

typedef struct _quad {
    int x1;
    int y1;
    int x2;
    int y2;

    int v;

    struct _quad* ul;
    struct _quad* ur;
    struct _quad* bl;
    struct _quad* br;
} quad;

int need_split(int x1, int y1, int x2, int y2) {
    int x,y;
    for (x=x1; x<x2; x++) {
        for (y=y1; y<y2; y++) {
            if (a[x][y] != a[x1][y1]) {
                return 1;
            }
        }
    }

    return 0;
}

void build(quad* n) {
    int x1 = n->x1;
    int x2 = n->x2;
    int xm = (x1+x2)/2;
    int y1 = n->y1;
    int y2 = n->y2;
    int ym = (y1+y2)/2;
    if (need_split(x1, y1, x2, y2)) {
        n->v = -1;
        quad *ul = malloc(sizeof(quad));
        ul->x1 = x1; ul->x2 = xm; ul->y1 = y1; ul->y2 = ym;
        n->ul = ul;
        build(ul);

        quad *ur = malloc(sizeof(quad));
        ur->x1 = xm; ur->x2 = x2; ur->y1 = y1; ur->y2 = ym;
        n->ur = ur;
        build(ur);

        quad *bl = malloc(sizeof(quad));
        bl->x1 = x1; bl->x2 = xm; bl->y1 = ym; bl->y2 = y2;
        n->bl = bl;
        build(bl);

        quad *br = malloc(sizeof(quad));
        br->x1 = xm; br->x2 = x2; br->y1 = ym; br->y2 = y2;
        n->br = br;
        build(br);
    } else {
        n->v = a[x1][y1];
        n->ul = NULL;
        n->ur = NULL;
        n->bl = NULL;
        n->br = NULL;
    }
}

void count(quad *node, int *b, int *w) {
    if (node == NULL) { return; }

    if (node->v < 0) { // node
        count(node->ul, b, w);
        count(node->ur, b, w);
        count(node->bl, b, w);
        count(node->br, b, w);
    } else {
        // leaf
        if (node->v == 1) { // blue
            *b = *b+1;
        } else {
            *w = *w+1;
        }
    }
}

int main() {
    int n,x,y,b,w;
    scanf("%d", &n);
    for (y=0; y<n; y++) {
        for (x=0; x<n; x++) {
            scanf("%d", &a[x][y]);
        }
    }

    quad root;
    root.x1 = 0;
    root.x2 = n;
    root.y1 = 0;
    root.y2 = n;
    build(&root);

    b = 0; w = 0;
    count(&root, &b, &w);

    printf("%d\n%d\n", w, b);

    return 0;
}


```
