```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define N 10001
int q[N];
int rear = -1, front = -1;

int isfull() {
    return (front == rear + 1) || (front == 0 && rear == N-1);
}

int isempty() {
    return front == -1;
}

int size() {
    if (isfull()) {
        return N;
    } else if (isempty()) {
        return 0;
    } else {
        return rear >= front ? rear - front + 1 : rear - front + N + 1;
    }
}

void enq(int elem) {
    if (isfull()) {
        // do nothing
    } else {
        if (front == -1) {
            front = 0;
        }
        rear = (rear + 1)%N;
        q[rear] = elem;
    }
}

int deq() {
    if (isempty()) {
        return -1;
    } else {
        int elem;
        elem = q[front];
        if (front == rear) {
            front = rear = -1;
        } else {
            front = (front+1)%N;
        }

        return elem;
    }
}

int main() {
    int n;
    char cmd[20];
    scanf("%d", &n);
    while ((getchar()) != '\n');
    while (n--) {
        fgets(cmd, 19, stdin);
        if (strncmp(cmd, "push", 4) == 0) {
            int x = atoi(cmd + 5);
            enq(x);
        } else if (strncmp(cmd, "pop", 3) == 0) {
            printf("%d\n", deq());
        } else if (strncmp(cmd, "size", 4) == 0) {
            printf("%d\n", size());
        } else if (strncmp(cmd, "empty", 5) == 0) {
            printf("%d\n", isempty());
        } else if (strncmp(cmd, "front", 5) == 0) {
            if (isempty()) {
                printf("-1\n");
            } else {
                printf("%d\n", q[front]);
            }
        } else if (strncmp(cmd, "back", 4) == 0) {
            if (isempty()) {
                printf("-1\n");
            } else {
                printf("%d\n", q[rear]);
            }
        }
    }

    return 0;
}
```
