비교연산을 함수로 뺌.

```
#include <stdio.h>
#include <string.h>

#define N (1<<17)
#define abs(x) ((x)>=0?(x):(-(x)))

int a[N] = {0,};
int sz = 0;

int isheap(int p, int c) {
    return abs(p) < abs(c) || (abs(p) == abs(c) && p < c);
}

void print() {
    int i;
    printf("heap = ");
    for (i=0; i<N; i++) {
        if (a[i] == 0) {
            break;
        }
        printf("%d, ", a[i]);
    }
    printf("\n");
}

void percolateDown(int i) {
    int li = 2*i+1, ri = 2*i+2, t;

    if (i >= sz || (a[li] == 0 && a[ri] == 0)) {
        return;
    }

    if ((a[li] != 0 && !isheap(a[i], a[li])) || (a[ri] != 0 && !isheap(a[i], a[ri]))) {
        if (a[li] != 0 && a[ri] != 0) {
            if (isheap(a[li], a[ri])) {
                t = a[li];
                a[li] = a[i];
                a[i] = t;
                percolateDown(li);
            } else {
                t = a[ri];
                a[ri] = a[i];
                a[i] = t;
                percolateDown(ri);
            }
        } else if (a[li] != 0) {
            t = a[li];
            a[li] = a[i];
            a[i] = t;
            percolateDown(li);
        } else if (a[ri] != 0) {
            t = a[ri];
            a[ri] = a[i];
            a[i] = t;
            percolateDown(ri);
        }
    }
}

void percolateUp(int i) {
    if (i == 0) {
        return;
    }
    int pi = (i-1)/2;
    // check is heap
    if (!isheap(a[pi], a[i])) {
        int t = a[i];
        a[i] = a[pi];
        a[pi] = t;
        percolateUp(pi);
    }
}

int pop() {
    if (sz <= 0) {
        return 0;
    }
    int t = a[0];
    a[0] = a[sz-1];
    a[sz-1] = 0;
    sz--;
    percolateDown(0);
    return t;
}

void insert(int x) {
    // insert to bin tree
    a[sz] = x;
    percolateUp(sz++);
}

int main() {
    int n, x;
    memset(a, 0, N*sizeof(int));

    scanf("%d", &n);
    while (n--) {
        scanf("%d", &x);

        if (x != 0) {
            insert(x);
        } else {
            printf("%d\n", pop());
        }
        // print();
    }

    return 0;
}
```
