#include <stdio.h>

int psum[100000];
int a[100000];

int main() {
    int N, M;
    int i, j;
    scanf("%d %d", &N, &M);
    for (i=0; i<N; i++) {
        scanf("%d", &a[i]);
    }
    psum[0] = a[0];
    for (i=1; i<N; i++) {
        psum[i] = psum[i-1] + a[i];
    }

    while (M--) {
        scanf("%d %d", &i, &j);
        printf("%d\n", psum[j-1] - psum[i-1] + a[i-1]);
    }

    return 0;
}
