거듭제곱의 분할정복

```
#include <stdio.h>

void mul(int (*a)[5], int (*b)[5], int (*c)[5], int sz) { // c = a * b
    int i,j,k;
    for (i=0; i<sz; i++) {
        for (j=0; j<sz; j++) {
            c[i][j] = 0;

            for (k=0; k<sz; k++) {
                c[i][j] += (a[i][k]*b[k][j])%1000;
            }

            c[i][j] %= 1000;
        }
    }
}

void mpow(int (*a)[5], int (*b)[5], long p, int n) { // b = a^p
    int i,j;
    int t[5][5];
    if (p == 1) {
        for (i=0; i<n; i++) {
            for (j=0; j<n; j++) {
                b[i][j] = a[i][j] % 1000;
            }
        }
        return;
    }

    if (p % 2) {
        mpow(a, t, p-1, n);
        mul(a, t, b, n);
    } else {
        mpow(a, t, p/2, n);
        mul(t, t, b, n);
    }
}

int main() {
    long p;
    int n, i, j;
    int a[5][5];
    int b[5][5];

    scanf("%d%ld", &n, &p);
    for (i=0; i<n; i++) {
        for (j=0; j<n; j++) {
            scanf("%d", &a[i][j]);
        }
    }

    mpow(a, b, p, n);

    for (i=0; i<n; i++) {
        for (j=0; j<n; j++) {
            printf("%d ", b[i][j]);
        }
        printf("\n");
    }

    return 0;
}
```
